			limit = { literacy_rate < 0.40 }

		}

			limit = { literacy_rate >= 0.60 }

			limit = { literacy_rate >= 0.80 }

		}

		if = {
			limit = {
				has_technology_researched = railways
			}
			multiply = 1.5
		}
	}
