ztr_demographic_0_to_1_progress_bar = {
	name = "ztr_demographic_0_to_1_progress_bar"
	desc = "ztr_demographic_0_to_1_progress_bar_desc"
	default_green = yes
	yearly_progress = {
		if = {
			limit = {
				OR = {
					NOT = {
						has_modifier = dev_0_country
					}
					year > 1900
				}
			}
			add = {
				desc = "ztr_demographic_0_to_1_development_tt"
				value = 0.3
			}
		}
		if = {
			limit = {
				average_sol > 10
			}
			add = {
				desc = "ztr_demographic_0_to_1_sol_tt"
				value = 0.4
			}
		}
		if = {
			limit = {
				literacy_rate > 0.3
			}
			add = {
				desc = "ztr_demographic_0_to_1_literacy_tt"
				value = 0.4
			}
		}
		if = {
			limit = {
				OR = {
					has_law_or_variant = law_type:law_private_health_insurance
					has_law_or_variant = law_type:law_public_health_insurance
				}
			}
			add = {
				desc = "ztr_demographic_healthcare_tt"
				value = 0.2
			}
		}
		if = {
			limit = {
				OR = {
					country_has_state_religion = rel:sunni
					country_has_state_religion = rel:shiite
					AND = {
						exists = c:DEI
						c:DEI = this
					}
					AND = {
						exists = c:BIC
						c:BIC = this
					}
				}
			}
			subtract = {
				desc = "ztr_demographic_traditional_values_tt"
				value = 0.05
			}
		}
	}
	start_value = 0
	min_value = 0
	max_value = 30
}

				desc = "ztr_demographic_slavery_tt"

			}
		}
		if = {
			limit = {
				is_in_geographic_region = geographic_region_north_america
			}
			subtract = {
				desc = "ztr_demographic_7_to_8_north_america_tt"
				value = 0.4
			}
		}
		if = {
			limit = {
				get_country_gdp_per > 25
				OR = {
					has_law_or_variant = law_type:law_multicultural
					has_law_or_variant = law_type:law_no_migration_controls
				}
				OR = {
					total_population < 200000000
					is_in_geographic_region = geographic_region_americas
				}
			}
			subtract = {
				desc = "ztr_demographic_7_to_8_reset_tt"
				value = 30
			}
		}
	}
	start_value = 0
	min_value = 0
	max_value = 30
}
