INJECT:engineers = {
	political_engagement_mult = {
		value = 1

		add = {
			desc = "POP_STARVATION"

			if = {
				limit = {
					is_in_mild_starvation = yes
				}
				value = 0.5
			}
			else_if = {
				limit = {
					is_in_severe_starvation = yes
				}
				value = 1.5
			}
		}

		multiply = {
			desc = "HAS_POLITICAL_AGITATION"
			value = 1.0

			# These two conditions are added in order to balance the rapid increase of engineers in late game
			if = {
				limit = {
					owner = {
						has_technology_researched = neoliberism
					}
				}
				value = 0.75 #1.5
			}
			else_if = {
				limit = {
					owner = {
						has_technology_researched = socialdemocracy
					}
				}
				value = 1
			}
			else_if = {
				limit = {
					owner = {
						has_technology_researched = political_agitation
					}
				}
				value = 1.25
			}
		}
	}
}
