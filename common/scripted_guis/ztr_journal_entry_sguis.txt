je_decolonization_colonial_states_sgui = {
	scope = country

	effect = {
		if = {
			limit = {
				has_variable_list = ztr_colonial_states_list
			}
			every_in_list = {
				variable = ztr_colonial_states_list
				custom_tooltip = "JE_DECOLONIZATION_COLONIAL_STATE_ENTRY"
			}
		}
		else = {
			custom_tooltip = "JE_DECOLONIZATION_NO_COLONIAL_STATES"
		}
	}
}
