		gas = { stance = wants_export }
	}


	possible = {

	}

	weight = {
		value = 10

		if = {
			limit = {
				OR = {
					is_in_geographic_region = geographic_region_europe
					is_in_geographic_region = geographic_region_north_america
				}
			}
			add = 20
		}

		if = {
			limit = {
				has_radical_leftist_government = yes
				NOT = {
					ig:ig_rural_folk ?= {
						is_powerful = yes
						is_in_government = yes
					}
				}
			}
			add = 25
		}

		if = {
			limit = { is_country_type = unrecognized }
			multiply = 0.5
		}

		if = {
			limit = {
				exists = ig:ig_industrialists
				ig:ig_industrialists ?= { is_powerful = yes }
			}
			multiply = 1.5
		}

		if = {
			limit = {
				has_technology_researched = railways
			}
			multiply = 2.0
		}

		if = {
			limit = {
				exists = overlord
				overlord = {
					OR = {
						AND = {
							is_ai = no
							NOT = { has_law_or_variant = law_type:law_traditionalism }
						}
						has_strategy = ai_strategy_resource_expansion
						has_strategy = ai_strategy_industrial_expansion
					}
				}
			}
			multiply = 2.0
		}
	}
}
