					is_building_type = building_urban_center
					level >= 50

# Fusion Reactor Journal Entries
je_fusion_reactor_research = {
	icon = "gfx/interface/icons/event_icons/innovation.dds"
	group = je_group_technology
	on_monthly_pulse = {
		effect = {
			# Progress depends on nation's development indicators
			change_variable = {
				name = fusion_reactor_var
				add = 0.25
			}
			# GDP per capita bonus: wealthy nations progress faster
			if = {
				limit = {
					gdp > 1000000000
				}
				change_variable = {
					name = fusion_reactor_var
					add = 0.25
				}
			}
			if = {
				limit = {
					gdp > 705032704
				}
				change_variable = {
					name = fusion_reactor_var
					add = 0.5
				}
			}
			# Literacy bonus: highly educated populace accelerates R&D
			if = {
				limit = {
					literacy_rate > 0.95
				}
				change_variable = {
					name = fusion_reactor_var
					add = 0.25
				}
			}
			# Major population bonus: large labor pool of scientists
			if = {
				limit = {
					total_population > 50000000
				}
				change_variable = {
					name = fusion_reactor_var
					add = 0.25
				}
			}
			if = {
				limit = {
					total_population > 500000000
				}
				change_variable = {
					name = fusion_reactor_var
					add = 0.25
				}
			}
		}
	}
	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}
	on_complete = {
		trigger_event = {
			id = ztr_technology.10
		}		# fusion reactor first breakthrough
	}
	invalid = {
		OR = {
			NOT = {
				has_technology_researched = smart_energy_grid
			}
			NOT = {
				has_technology_researched = third_nuclear_reactor
			}
			NOT = {
				has_research_center = yes
			}
		}
	}
	on_invalid = {
		remove_variable = fusion_reactor_var
	}
	current_value = {
		value = root.var:fusion_reactor_var
	}
	goal_add_value = {
		value = 100
	}
	progressbar = yes
	weight = 10
	should_be_pinned_by_default = yes
	modifiers_while_active = {
		fusion_reactor_research
	}
}

# IMF Modernization Journal Entry
je_imf_modernization = {
	icon = "gfx/interface/icons/event_icons/academia.dds"
	group = je_group_technology
	scripted_button = ztr_imf_request_technical_mission
	scripted_button = ztr_imf_launch_development_plan
	scripted_button = ztr_imf_seek_scholarships
	on_yearly_pulse = {
		random_events = {
			200 = 0
			10 = ztr_imf_modernization.10			# Brain Drain Crisis
			10 = ztr_imf_modernization.11			# IMF Austerity Conditions
			10 = ztr_imf_modernization.12			# Foreign Investment Boom
			10 = ztr_imf_modernization.13			# Rural Education Campaign
			10 = ztr_imf_modernization.14			# Political Instability Dilemma
			10 = ztr_imf_modernization.15			# International Aid Conference
		}
	}
	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_imf_modernization_status_advanced
				trigger = {
					literacy_rate >= 0.6
					average_sol >= 16
					get_country_gdp_per > 5
				}
			}
			triggered_desc = {
				desc = je_imf_modernization_status_progressing
				trigger = {
					literacy_rate >= 0.45
					average_sol >= 13
					get_country_gdp_per > 3
				}
			}
			triggered_desc = {
				desc = je_imf_modernization_status_early
				trigger = {
					literacy_rate >= 0.3
					average_sol >= 10
				}
			}
			triggered_desc = {
				desc = je_imf_modernization_status_initial
				trigger = {
					always = yes
				}
			}
		}
	}
	complete = {
		custom_tooltip = {
			text = je_imf_modernization_literacy_tt
			literacy_rate >= 0.7
		}
		custom_tooltip = {
			text = je_imf_modernization_sol_tt
			average_sol >= 18
		}
		custom_tooltip = {
			text = je_imf_modernization_developed_tt
			get_country_gdp_per > 7
		}
	}
	on_complete = {
		set_variable = {
			name = imf_modernization_completed
			value = 1
		}
		trigger_event = ztr_imf_event.5
	}
	modifiers_while_active = {
		imf_modernization_commitment_modifier
	}
	weight = 25
	should_be_pinned_by_default = yes
}
