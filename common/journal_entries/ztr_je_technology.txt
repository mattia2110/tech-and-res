je_research_center_site = {
	icon = "gfx/interface/icons/event_icons/innovation.dds"

	group = je_group_technology

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = research_center_var
				add = 1
			}
		}
	}

	complete = {
		scope:journal_entry = { is_goal_complete = yes }
		# There should be a state like that, despite research center will always be built in capital
		any_scope_state = {
			is_incorporated = yes
			is_split_state = no
			any_scope_building = {
				is_building_type = building_urban_center
				level >= 50
			}
			any_scope_building = {
				is_building_type = building_university
				level >= 5
			}
		}
	}

	on_complete = {
		#remove_modifier = research_center_survey
		trigger_event = { id = ztr_technology.1 } # site found
	}

	invalid = {
		NOT = {
			any_scope_state = {
				is_incorporated = yes
				is_split_state = no
				any_scope_building = {
					is_building_type = building_urban_center
					level >= 50
				}
				any_scope_building = {
					is_building_type = building_university
					level >= 5
				}
			}
		}
	}

	on_invalid = {
		remove_modifier = research_center_survey
		remove_variable = research_center_var
	}

	current_value = {
		value = root.var:research_center_var
	}

	goal_add_value = {
		value = 24
	}

	progressbar = yes

	weight = 10

	should_be_pinned_by_default = yes
}

# Fusion Reactor Journal Entries

je_fusion_reactor_research = {
	icon = "gfx/interface/icons/event_icons/innovation.dds"

	group = je_group_technology

	on_monthly_pulse = {
		effect = {
			# Progress depends on nation's development indicators
			change_variable = {
				name = fusion_reactor_var
				add = 0.25
			}

			# GDP per capita bonus: wealthy nations progress faster
			if = {
				limit = { gdp > 1000000000 }
				change_variable = {
					name = fusion_reactor_var
					add = 0.25
				}
			}
			if = {
				limit = { gdp > 5000000000 }
				change_variable = {
					name = fusion_reactor_var
					add = 0.5
				}
			}

			# Literacy bonus: highly educated populace accelerates R&D
			if = {
				limit = { literacy_rate > 0.95 }
				change_variable = {
					name = fusion_reactor_var
					add = 0.25
				}
			}

			# Major population bonus: large labor pool of scientists
			if = {
				limit = { total_population > 50000000 }
				change_variable = {
					name = fusion_reactor_var
					add = 0.25
				}
			}
			if = {
				limit = { total_population > 500000000 }
				change_variable = {
					name = fusion_reactor_var
					add = 0.25
				}
			}
		}
	}

	complete = {
		scope:journal_entry = { is_goal_complete = yes }
	}

	on_complete = {
		trigger_event = { id = ztr_technology.10 } # fusion reactor first breakthrough
	}

	invalid = {
		OR = {
			NOT = {
				has_technology_researched = smart_energy_grid
			}
			NOT = {
				has_technology_researched = third_nuclear_reactor
			}
			NOT = {
				has_research_center = yes
			}
		}
	}

	on_invalid = {
		remove_variable = fusion_reactor_var
	}

	current_value = {
		value = root.var:fusion_reactor_var
	}

	goal_add_value = {
		value = 100
	}

	progressbar = yes

	weight = 10

	should_be_pinned_by_default = yes

	modifiers_while_active = {
		fusion_reactor_research
	}
}
