
je_ottoman_empire_collapse = {
	icon = "gfx/interface/icons/event_icons/waving_flag_ottoman.dds"
	group = je_group_global_historical_content
	is_shown_when_inactive = {
		has_game_rule = allow_ztr_historical_content
		year > 1900
		exists = c:TUR
	}
	should_be_involved = {
		OR = {
			c:TUR ?= this
			AND = {
				country_rank = rank_value:great_power
				navy_size >= 30 #Naval Power
			}
		}
	}
	is_shown_in_lobby = {
		c:TUR ?= this
		is_involved_in_journal_entry = je_ottoman_empire_collapse
	}

	scripted_button = ztr_ottoman_collapse_request_egypt
	scripted_button = ztr_ottoman_collapse_request_levant
	scripted_button = ztr_ottoman_collapse_request_iraq
	possible = {
		has_game_rule = allow_ztr_historical_content
		exists = c:TUR
		c:TUR = {
			# Power Humiliation
			NOT = {
				country_rank = rank_value:great_power
			}
			# Autocratic and Low SOL
			ztr_is_not_democratic = yes
			average_sol < 12
		}
	}
	immediate = {
		c:TUR = {
			save_scope_as = ztr_ottoman_collapse_target_country
			set_variable = {
				name = ottoman_empire_collapse_var
				value = 0
			}
			set_variable = {
				name = ottoman_empire_collapse_arab_var
				value = 0
			}
			set_variable = {
				name = ottoman_empire_collapse_army_var
				value = 0
			}
			set_variable = {
				name = ottoman_empire_collapse_foreign_var
				value = 0
			}
		}
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					NOT = {
						has_global_variable = ztr_ottoman_collapse_happened
					}
				}
				set_global_variable = ztr_ottoman_collapse_happened
				# We can only accept or refuse the invitation once
				scope:journal_entry = {
					every_scope_je_involved = {
						limit = {
							NOT = {
								c:TUR ?= this
							}
						}
						trigger_event = {
							id = ztr_ottoman_collapse.100
							days = 5
							popup = yes
						}
					}
				}
				c:TUR = {
					trigger_event = {
						id = ztr_ottoman_collapse.1
						days = 30
						popup = yes
					}
				}
			}
		}
	}
	on_yearly_pulse = {
		effect = {
			scope:ztr_ottoman_collapse_target_country = {
				change_variable = {
					name = ottoman_empire_collapse_var
					add = 2
				}
				# Liberal reforms lessen centrifugal forces
				if = {
					limit = {
						has_law = law_type:law_universal_suffrage
					}
					change_variable = {
						name = ottoman_empire_collapse_var
						subtract = 1
					}
				}
				# More inclusive citizenship reduces provincial separatism
				if = {
					limit = {
						has_law = law_type:law_cultural_exclusion
					}
					change_variable = {
						name = ottoman_empire_collapse_var
						subtract = 1
					}
				}
				if = {
					limit = {
						has_law = law_type:law_multicultural
					}
					change_variable = {
						name = ottoman_empire_collapse_var
						subtract = 2
					}
				}
				# Economic weakness fuels decline
				if = {
					limit = {
						get_country_gdp_per < 2.5
					}
					change_variable = {
						name = ottoman_empire_collapse_var
						add = 1
					}
				}
				# Russia's ambitions for the straits and eastern Anatolia
				if = {
					limit = {
						c:RUS = {
							exists = yes
							country_rank = rank_value:great_power
						}
					}
					change_variable = {
						name = ottoman_empire_collapse_var
						add = 1
					}
				}
				# Slipping down the prestige ladder accelerates disintegration
				if = {
					limit = {
						country_rank < rank_value:major_power
					}
					change_variable = {
						name = ottoman_empire_collapse_var
						add = 2
					}
				}
				# Crisis fronts that have been mishandled for several years now feed the main collapse meter
				if = {
					limit = {
						var:ottoman_empire_collapse_arab_var >= 4
					}
					change_variable = {
						name = ottoman_empire_collapse_var
						add = 1
					}
				}
				if = {
					limit = {
						var:ottoman_empire_collapse_army_var >= 4
					}
					change_variable = {
						name = ottoman_empire_collapse_var
						add = 1
					}
				}
				if = {
					limit = {
						var:ottoman_empire_collapse_foreign_var >= 4
					}
					change_variable = {
						name = ottoman_empire_collapse_var
						add = 1
					}
				}
				random_list = {
					20 = {
					}
					5 = {
						trigger_event = {
							id = ztr_ottoman_collapse.2
							days = 20
						}
					}
					5 = {
						trigger_event = {
							id = ztr_ottoman_collapse.3
							days = 50
						}
					}
					5 = {
						trigger_event = {
							id = ztr_ottoman_collapse.4
							days = 90
						}
					}
				}
			}
		}
