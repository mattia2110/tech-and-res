je_demographic_0_to_1 = {
	icon = "gfx/interface/icons/event_icons/medical_degrees.dds"
	group = je_group_technology
	should_be_pinned_by_default = yes

	is_shown_when_inactive = {
	}

	possible = {
	}

	scripted_progress_bar = ztr_demographic_0_to_1_progress_bar

	complete = {
		custom_tooltip = {
			text = ztr_demographic_stage_complete_tt
			scope:journal_entry ?= {
				"scripted_bar_progress(ztr_demographic_0_to_1_progress_bar)" >= 30
			}
		}
	}

	on_complete = {
		remove_state_demographic_modifier = yes
		add_modifier = demographic_stage_1
		add_journal_entry = {
			type = je_demographic_1_to_2
		}
	}

	transferable = no

	weight = 1
}
