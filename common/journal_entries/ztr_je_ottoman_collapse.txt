                limit = { get_country_gdp_per < 1.5 }

                    name = ottoman_empire_collapse_var
                    add = 1
                }

                limit = {
                    c:RUS = {
						exists = yes
						country_rank = rank_value:great_power
					}
                }
                change_variable = {
                    name = ottoman_empire_collapse_var
                    add = 1
                }

            # Slipping down the prestige ladder

				limit = {
					country_rank < rank_value:great_power
				}

					name = ottoman_empire_collapse_var
					add = 1
				}
			}
