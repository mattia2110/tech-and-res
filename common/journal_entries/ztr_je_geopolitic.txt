# Liberty Index
je_liberty_index = {
	icon = "gfx/interface/icons/event_icons/ideologies.dds"
	group = je_group_historical_content
	is_shown_when_inactive = {
		technres_is_active = yes
	}
	possible = {
		#has_technology_researched = human_rights
	}
	immediate = {
		set_variable = {
			name = liberty_index_var
			value = 0
		}
		set_variable = {
			name = equality_index_var
			value = 0
		}
	}
	scripted_button = show_liberty_index_rankings_button
	scripted_button = show_equality_index_rankings_button
	scripted_progress_bar = liberty_index_progress_bar
	scripted_progress_bar = equality_index_progress_bar
	on_yearly_pulse = {
		effect = {
			# Calculate Liberty and Equality Index
			if = {
				limit = {
					BPM_is_active_trigger = yes
				}
				calculate_liberty_equality_index_bpm = yes
			}
			else = {
				calculate_liberty_equality_index = yes
			}
			calculate_max_liberty_index = yes
			scope:journal_entry = {
				set_bar_progress = {
					value = root.var:liberty_index_var
					name = liberty_index_progress_bar
				}
				set_bar_progress = {
					value = root.var:equality_index_var
					name = equality_index_progress_bar
				}
			}
		}
	}
	transferable = no
	weight = 1
}

# Superpowr ask diplomatically to another country to change laws
# Always use with:
# - ztr_has_superpower_alignment_law_candidate scripted trigger
je_superpower_request_law_enactment = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"
	group = je_group_technology
	timeout = 1825
	immediate = {
		random_country = {
			limit = {
				has_variable = superpower_requester_flag
			}
			save_scope_as = superpower_law_requester
		}
		ztr_select_superpower_alignment_law = yes
	}
	on_yearly_pulse = {
		effect = {
			if = {
				limit = {
					is_ai = yes
					exists = scope:superpower_alignment_law
					NOT = {
						has_law_or_variant = scope:superpower_alignment_law.type
					}
					enacting_any_law = no
					scope:superpower_alignment_law = {
						law_is_available = yes
					}
				}
				scope:superpower_alignment_law = {
					ROOT = {
						start_enactment = prev.type
					}
				}
			}
		}
	}
	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_superpower_request_law_enactment_status_enacting
				trigger = {
					is_enacting_law = scope:superpower_alignment_law.type
				}
			}
			triggered_desc = {
				desc = je_superpower_request_law_enactment_status_pending
				trigger = {
					always = yes
				}
			}
		}
	}
	complete = {
		has_law_or_variant = scope:superpower_alignment_law.type
	}
	on_complete = {
		if = {
			limit = {
				exists = scope:superpower_law_requester
			}
			change_relations = {
				country = scope:superpower_law_requester
				value = 50
			}
			custom_tooltip = {
				text = je_superpower_aid_extended_tt
				remove_modifier = superpower_aid_modifier
				add_modifier = {
					name = superpower_aid_modifier
					years = 10
				}
			}
		}
	}
	invalid = {
		OR = {
			NOT = {
				exists = scope:superpower_law_requester
			}
			scope:superpower_law_requester ?= {
				is_country_alive = no
			}
			NOT = {
				exists = scope:superpower_alignment_law
			}
		}
	}
	on_timeout = {
		if = {
			limit = {
				exists = scope:superpower_law_requester
			}
			remove_modifier = superpower_aid_modifier
			change_relations = {
				country = scope:superpower_law_requester
				value = -150
			}
		}
	}
	event_outcome_completed_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_superpower_request_law_enactment_completed_desc
			}
		}
	}
	event_outcome_timeout_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_superpower_request_law_enactment_timeout_desc
			}
		}
	}
	weight = 25
	should_be_pinned_by_default = yes
}

je_un_votations = {
	icon = "gfx/interface/icons/event_icons/united_nations.dds"
	group = je_group_global_un_content
	is_shown_when_inactive = {
		has_global_variable = ztr_un_vote_active
	}
	should_be_involved = {
		has_variable = ztr_un_vote_eligible
	}
	should_show_when_not_involved = {
		OR = {
			has_modifier = ztr_onu_member
			has_modifier = ztr_onu_permanent_member
		}
	}
	is_shown_in_lobby = {
		has_global_variable = ztr_un_vote_active
		OR = {
			has_modifier = ztr_onu_member
			has_modifier = ztr_onu_permanent_member
		}
	}
	possible = {
		has_global_variable = ztr_un_vote_active
	}
	immediate = {
		scope:journal_entry = {
			ztr_un_vote_update_results = yes
		}
	}
	on_monthly_pulse = {
		effect = {
			scope:journal_entry = {
				ztr_un_vote_update_results = yes
			}
			# Countdown: decrement months remaining
			if = {
				limit = {
					has_global_variable = ztr_un_vote_months_remaining
					global_var:ztr_un_vote_months_remaining > 0
				}
				change_global_variable = {
					name = ztr_un_vote_months_remaining
					add = -1
				}
			}
			# Vote session ended: resolve and deactivate
			if = {
				limit = {
					has_global_variable = ztr_un_vote_months_remaining
					global_var:ztr_un_vote_months_remaining <= 0
				}
				scope:journal_entry = {
					ztr_un_vote_update_results = yes
				}
				# Check if vote passes: support > against AND no veto blocks it
				if = {
					limit = {
						global_var:ztr_un_vote_support_count > global_var:ztr_un_vote_against_count
						# Veto check: if veto enabled, no permanent member may have voted against
						OR = {
							NOT = {
								global_var:ztr_un_law_veto_right = 1
							}
							NOT = {
								any_country = {
									has_modifier = ztr_onu_permanent_member
									has_variable = ztr_un_vote_choice_against
								}
							}
						}
					}
					# Vote passed - apply the governance change
					ztr_un_apply_governance_vote = yes
				}
				else = {
					# Vote failed (majority lost or vetoed) - clean up subject/proposal
					if = {
						limit = {
							has_global_variable = ztr_un_vote_subject
						}
						remove_global_variable = ztr_un_vote_subject
					}
					if = {
						limit = {
							has_global_variable = ztr_un_vote_proposal
						}
						remove_global_variable = ztr_un_vote_proposal
					}
				}
				# Cleanup removes ztr_un_vote_active, making possible = no, then JE deactivates
				ztr_un_vote_cleanup_all_members = yes
			}
			# Edge case: no eligible countries left, then deactivate
			else_if = {
				limit = {
					NOT = {
						any_country = {
							OR = {
								has_variable = ztr_un_vote_eligible
								is_involved_in_journal_entry = je_un_votations
							}
						}
					}
				}
				ztr_un_vote_cleanup_all_members = yes
			}
		}
	}
	status_desc = {
		first_valid = {
			triggered_desc = {
				desc = je_un_votations_status_support
				trigger = {
					global_var:ztr_un_vote_support_count > global_var:ztr_un_vote_against_count
				}
			}
			triggered_desc = {
				desc = je_un_votations_status_against
				trigger = {
					global_var:ztr_un_vote_against_count > global_var:ztr_un_vote_support_count
				}
			}
			triggered_desc = {
				desc = je_un_votations_status_neutral
				trigger = {
					always = yes
				}
			}
		}
	}
	scripted_progress_bar = ztr_un_vote_support_progress_bar
	scripted_progress_bar = ztr_un_vote_neutral_progress_bar
	scripted_progress_bar = ztr_un_vote_against_progress_bar
	scripted_button = ztr_un_vote_for_statement
	scripted_button = ztr_un_vote_neutral_statement
	scripted_button = ztr_un_vote_against_statement
	weight = 110
	should_be_pinned_by_default = yes
	can_deactivate = yes
	transferable = no
	progressbar = no
}
