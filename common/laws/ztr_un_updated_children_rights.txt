# Children's Rights (BPM seems to be vanilla in this law group)

REPLACE:law_child_labor_allowed = {
	group = lawgroup_childrens_rights

	icon = "gfx/interface/icons/law_icons/child_labor_allowed.dds"

	progressiveness = -50

	#T&R
	can_enact = {
		custom_tooltip = {
			text = ztr_un_hr_restricted_labor_children_rights_tt
			OR = {
				NOT = { ztr_is_un_member = yes }
				NOT = {
					global_var:ztr_un_hr_children_rights > 0
				}
			}
		}
	}

	modifier = {
		state_laborers_mortality_mult = 0.05
		state_machinists_mortality_mult = 0.05
		state_farmers_mortality_mult = 0.05
		state_peasants_mortality_mult = 0.05

		state_dependent_wage_mult = 0.30
	}

	ai_will_do = {
		exists = ruler
		ruler = {
			has_ideology = ideology:ideology_market_liberal
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}
