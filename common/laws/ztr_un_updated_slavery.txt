# Slavery (BPM compatible as possible)

REPLACE:law_colonial_slavery = {
	group = lawgroup_slavery
	progressiveness = 50
	icon = "gfx/interface/icons/law_icons/colonial_slavery.dds"
	can_enact = {
		any_scope_state = {
			is_incorporated = no
		}
		#T&R
		custom_tooltip = {
			text = ztr_un_hr_illegalized_slavery_tt
			ztr_is_un_member = yes
			NOT = {
				has_global_variable = ztr_un_hr_slavery
				global_var:ztr_un_hr_slavery = 1
			}
		}
	}
	on_activate = {
		liberate_slaves_in_incorporated_states = yes
		if = {
			limit = {
				has_variable = slavery_recently_abolished
			}
			remove_variable = slavery_recently_abolished
		}
		hidden_effect = {
			if = {
				limit = {
					any_political_movement = {
						is_political_movement_type = movement_pro_slavery
						has_core_ideology = ideology:ideology_pro_slavery_movement
					}
				}
				random_political_movement = {
					limit = {
						is_political_movement_type = movement_pro_slavery
						has_core_ideology = ideology:ideology_pro_slavery_movement
					}
					set_core_ideology = ideology_pro_slavery_colonial_movement
				}
			}
		}
	}
	ai_enact_weight_modifier = {
		value = 0
		#Brazilian AI is more likely to pursue colonial slavery if there are no abolitionists in power.
		if = {
			limit = {
				c:BRZ ?= this
				has_law_or_variant = law_type:law_slave_trade
				NOR = {
					any_interest_group = {
						is_in_government = yes
						leader = {
							has_ideology = ideology:ideology_abolitionist
						}
					}
					ruler ?= {
						has_ideology = ideology:ideology_abolitionist
					}
					heir ?= {
						has_ideology = ideology:ideology_abolitionist
					}
				}
				scope:law.type = law_type:law_colonial_slavery
			}
			add = 250
		}
		# Brazilian AI is more likely to pursue colonial slavery if the Aberdeen Act is being enforced
		if = {
			limit = {
				c:BRZ ?= this
				has_law_or_variant = law_type:law_slave_trade
				has_modifier = aberdeen_act_enforced_brz
				scope:law.type = law_type:law_colonial_slavery
			}
			add = 250
		}
		#Petitions
		if = {
			limit = {
				ai_has_enact_weight_modifier_journal_entries = yes
			}
			add = 750
		}
	}
}

REPLACE:law_debt_slavery = {
	group = lawgroup_slavery
	progressiveness = -50
	icon = "gfx/interface/icons/law_icons/debt_slavery.dds"
	disallowing_laws = {
		law_multicultural
		law_affirmative_action
	}
	#T&R
	can_enact = {
		custom_tooltip = {
			text = ztr_un_hr_illegalized_slavery_tt
			ztr_is_un_member = yes
			NOT = {
				has_global_variable = ztr_un_hr_slavery
				global_var:ztr_un_hr_slavery = 1
			}
		}
		#BPM Compatibility
		trigger_if = {
			limit = {
				BPM_is_active_trigger = yes
			}
			trigger_if = {
				limit = {
					ROOT = c:USA
				}
				NOT = { has_modifier = bpm_gag_rule }
				NOT = { 
					has_modifier = bpm_constitutional_issues 
				}
			}
			trigger_else_if = {
				limit = {
					ROOT = c:BRZ
					root = { 
						has_dlc_feature = rp1_content
						is_player = yes
					}
				}
				NOT = { has_modifier = unrestricted_slave_trade_modifier }
			}
		}
	}
	on_activate = {
		#BPM Compatibility
		if = {
			limit = {
				BPM_is_active_trigger = yes
			}
			ig:ig_landowners ?= {
				remove_modifier = bpm_landowner_wary
			}
			ig:ig_reactionaries ?= {
				remove_modifier = bpm_landowner_wary
			}
		}
		else = {
			set_variable = {
				name = slavery_recently_enacted
				value = yes
				days = 1825
			}
			every_scope_state = {
				limit = {
					NOT = {
						any_scope_pop = {
							is_pop_type = slaves
						}
					}
				}
				set_variable = {
					name = former_free_state
					value = yes
					days = 1825
				}
			}
			ig:ig_landowners ?= {
				switch_from_colonial_slavery_to_pro_slavery = yes
			}
			hidden_effect = {
				if = {
					limit = {
						any_political_movement = {
							is_political_movement_type = movement_pro_slavery
							has_core_ideology = ideology:ideology_pro_slavery_colonial_movement
						}
					}
					random_political_movement = {
						limit = {
							is_political_movement_type = movement_pro_slavery
							has_core_ideology = ideology:ideology_pro_slavery_colonial_movement
						}
						set_core_ideology = ideology_pro_slavery_movement
					}
				}
			}
		}		
	}
	modifier = {
		interest_group_ig_landowners_pol_str_mult = 0.25
	}
	# AI should never enact debt slavery
	ai_will_do = {
		always = no
	}
	ai_enact_weight_modifier = {
		#Petitions
		value = 0
		if = {
			limit = {
				ai_has_enact_weight_modifier_journal_entries = yes
			}
			add = 750
		}
	}
}
