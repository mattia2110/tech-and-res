law_uranium_civil_usage = {
	group = lawgroup_uranium_usage
	
	icon = "gfx/interface/icons/law_icons/law_uranium_civil_usage.dds"
	
	progressiveness = 0

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}

	on_activate = {
		custom_tooltip = {
			text = nuclear_weapons_silo_destroy_tt
			every_scope_state = {
				limit = { has_modifier = nuclear_weapon_site}
				if = {
					limit = { has_building = building_nuclear_weapons_silo }
					remove_building = building_nuclear_weapons_silo
				}
			}
		}
	}
}
