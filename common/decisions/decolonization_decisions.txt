decolonization_tunisia_decision = {

	is_shown = {
		has_global_variable = ztr_decolonization_age_var
		NOT = { is_subject = yes }
		NOT = {
			OR = {
				c:TUN ?= root
			}
		}
		NOT = {
			country_has_primary_culture = cu:maghrebi
		}
		s:STATE_TUNISIA = {
			any_scope_state = {
				owner = ROOT
			}
		}
		NOT = {
			c:ITA ?= root
		}
		has_game_rule = allow_ztr_decolonization_content
	}

	possible = {
	}

	when_taken = {
		change_infamy = -5
		if = {
			limit = {
				NOT = {
				exists = c:TUN
			}
			}
			create_country = {
			tag = TUN
			origin = this
			province = p:x80D0A0
			on_created = {
				set_country_type = recognized
				activate_law = law_type:law_parliamentary_republic
				activate_law = law_type:law_autocracy
			}
			}
		}
		s:STATE_TUNISIA = {
			random_scope_state = {
				limit = {
					owner = root
				}
			set_state_owner = c:TUN
			}
		}
		create_bidirectional_truce = { country = c:TUN months = 120 }
	}

	ai_chance = {
		value = 5	
	}
}
