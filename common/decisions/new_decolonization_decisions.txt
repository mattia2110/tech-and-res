# Africa

decolonization_kenya_decision = {

	is_shown = {
		has_global_variable = ztr_decolonization_age_var
		NOT = { is_subject = yes }
		any_scope_state = {
			NOT = { is_homeland_of_country_cultures = root }
			state_region = {
				OR = {
					region = sr:region_zanj
					region = sr:region_ethiopia
				}
				is_homeland = cu:kikuyu
			}
			count > 0
		}
		OR = {
			has_state_in_state_region = STATE_KENYA
			has_state_in_state_region = STATE_RIFT_VALLEY
		}
		has_game_rule = allow_ztr_decolonization_content
	}

	possible = {
	}

	when_taken = {
		change_infamy = -5
		ztr_define_kenya = yes
		create_bidirectional_truce = { country = c:ZKN months = 120 }
		c:ZKN = {
			add_journal_entry = {
				type = je_demographic_0_to_1
			}
		}
	}

	ai_chance = {
		value = 5	
	}
}
