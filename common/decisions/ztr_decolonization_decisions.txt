good_friday_agreement_decision = {

	is_shown = {
		year > 1980
		exists = c:IRE
        THIS = c:IRE

	}

		any_scope_state = {

            }
		}

	when_taken = {
		c:GBR = {
			trigger_event = irish_independence.7
		}
	}

		value = 100	
	}
}

			overlord = root
		}

		}

		}
		OR = {
			NOT = { exists = c:KAL }

		}
		OR = {
			NOT = { exists = c:MAK }

		has_game_rule = allow_ztr_historical_content

	}

		set_global_variable = {

			value = 0
		}
		root = { trigger_event = durrand_line_event.1 }

indian_unifier_decision = {
	is_shown = {
		has_game_rule = allow_ztr_decolonization_content
		has_technology_researched = pan-nationalism
		country_rank >= rank_value:minor_power # In this way dwarf state will not unify India
		year > 1944

			AND = {
				exists = c:BHT

			name = ztr_indian_unifier_var
			value = 0
		}

		value = 100	
	}

# India can adopt Hinduism as state religion if Religion has an important role
ztr_hindu_decision = {
	is_shown = {
		c:BHT ?= this
		NOT = {
			country_has_state_religion = rel:hindu
		}
	}

	possible = {
		ig:ig_devout ?= {
			is_marginal = no
		}
		NOR = {
			has_law_or_variant = law_type:law_total_separation
			has_law_or_variant = law_type:law_state_atheism
		}
	}

	when_taken = {
		set_state_religion = rel:hindu
		state_religion_switch_effect = yes
		india_independence_ig_effects = yes
		random_scope_state = {
			limit = {
				owner = ROOT
				is_capital = yes
			}
			convert_population = {
				target = rel:hindu
				value = 0.7
			}
		}
		every_scope_state = {
			limit = {
				owner = ROOT
				is_incorporated = yes
				NOT = { is_capital = yes }
			}
			convert_population = {
				target = rel:hindu
				value = 0.3
			}
		}
	}

	ai_chance = {
		value = 20
	}
}

decolonization_uae_decision = {

	is_shown = {
		has_global_variable = ztr_decolonization_age_var
		cu:bedouin = { wants_to_be_decolonized = yes }
		NOT = { is_subject = yes }
		NOT = {
			country_has_primary_culture = cu:bedouin
		}
		NOT = {
			OR = {
				c:BHN ?= root
				c:ABU ?= root
			}
		}
		OR = {
			s:STATE_ABU_DHABI = {
			any_scope_state = {
					owner = ROOT
				}
			}
		}
		has_game_rule = allow_ztr_decolonization_content
	}

	possible = {
		cu:bedouin = { wants_to_be_decolonized = yes }
	}

	when_taken = {
		change_infamy = -5
		if = {
			limit = {
				NOT = {
					exists = c:BHN
				}
			}
			create_country = {
			tag = BHN
			origin = this
			province = p:x505AE0
			on_created = {
				set_country_type = recognized
				activate_law = law_type:law_theocracy
			}
			}
		}
		if = {
			limit = {
				NOT = {
					exists = c:ABU
				}
			}
			create_country = {
			tag = ABU
			origin = this
			province = p:x65a581
			on_created = {
				set_country_type = recognized
				activate_law = law_type:law_theocracy
			}
			}
		}
		if = {
			limit = {
				s:STATE_ABU_DHABI = {
				any_scope_state = {
						owner = ROOT
					}
				}
			}
			s:STATE_ABU_DHABI = {
				set_owner_of_provinces = {
					country = c:BHN
					provinces = { x505AE0 x2768F8 xEF5020}
				}
				random_scope_state = {
					limit = {
						owner = root
					}
					set_state_owner = c:ABU
				}
			}
		}
		create_bidirectional_truce = { country = c:BHN months = 120 }
		create_bidirectional_truce = { country = c:ABU months = 120 }
	}

	ai_chance = {
		value = 5	
	}
}

decolonization_india_decision = {

	is_shown = {
		has_game_rule = allow_ztr_decolonization_content		
		NOT = { has_variable = ztr_indian_unifier_var }
		any_scope_culture = {
			shares_heritage_trait_group_with_culture = cu:bengali #Indian
			this = { wants_to_be_decolonized = yes }
		}
		# exists = scope:ztr_indian_unifier_scope
		# NOT = { scope:ztr_indian_unifier_scope = root }
		#NOT = { c:PAK ?= root } #Pakistan future addition
		#NOT = { c:BGD ?= root } #Bangladesh future addition
		OR = {
			AND = {
				has_technology_researched = pan-nationalism
				country_is_in_india = yes
			}
			AND = {
				has_global_variable = ztr_indian_unifier_global_var
				NOT = { country_is_in_india = yes }
			}
		}
		NOT = { is_subject = yes }
		any_scope_state = {
			state_is_in_india = yes
			NOT = { state_region = s:STATE_CEYLON }
		}
	}

	possible = {
	}

	when_taken = {
		change_infamy = -10
		if = {
			limit = {
				exists = c:BHT
				c:BHT = { has_variable = ztr_indian_unifier_var }
			}
			every_scope_state = {
				limit = {
					state_is_in_india = yes
					NOT = { state_region = s:STATE_CEYLON }
				}
				set_state_owner = c:BHT
			}
		}
		if = {
			limit = {
				country_is_in_europe = yes 
			}
			create_bidirectional_truce = { country = scope:ztr_indian_unifier_scope months = 120 }
		}
	}

	ai_chance = {
		value = 5	
	}
}
