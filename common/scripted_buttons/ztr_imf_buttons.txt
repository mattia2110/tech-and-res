# IMF Modernization Buttons
ztr_imf_request_technical_mission = {
	name = "ztr_imf_request_technical_mission"
	desc = "ztr_imf_request_technical_mission_desc"
	visible = {
		has_journal_entry = je_imf_modernization
	}
	possible = {
		custom_tooltip = {
			text = "ztr_imf_technical_mission_cooldown_tt"
			NOT = {
				has_variable = imf_technical_mission_cooldown
			}
		}
	}
	effect = {
		set_variable = {
			name = imf_technical_mission_cooldown
			years = 10
		}
		custom_tooltip = ztr_imf_technical_mission_cost_tt
		trigger_event = ztr_imf_modernization.20
	}
	ai_chance = {
		value = 0
		if = {
			limit = {
				literacy_rate < 0.55
			}
			add = 3
		}
		if = {
			limit = {
				get_country_gdp_per < 5
			}
			add = 2
		}
		if = {
			limit = {
				scaled_debt > 0.25
			}
			multiply = 0
		}
	}
}

ztr_imf_launch_development_plan = {
	name = "ztr_imf_launch_development_plan"
	desc = "ztr_imf_launch_development_plan_desc"
	visible = {
		has_journal_entry = je_imf_modernization
		NOT = {
			has_variable = imf_development_plan_launched
		}
	}
	possible = {
		hidden_trigger = {
			NOT = {
				has_variable = imf_development_plan_launched
			}
		}
	}
	effect = {
		set_variable = imf_development_plan_launched
		add_modifier = {
			name = imf_development_plan_modifier
			years = 20
		}
	}
	ai_chance = {
		value = 2
		if = {
			limit = {
				average_sol >= 14
			}
			add = 5
		}
		if = {
			limit = {
				get_country_gdp_per < 7
			}
			add = 2
		}
		if = {
			limit = {
				ztr_is_not_democratic = yes
			}
			add = 2
		}
		if = {
			limit = {
				scaled_debt > 0.25
			}
			multiply = 0
		}
	}
}

ztr_imf_seek_scholarships = {
	name = "ztr_imf_seek_scholarships"
	desc = "ztr_imf_seek_scholarships_desc"
	visible = {
		has_journal_entry = je_imf_modernization
	}
	possible = {
		custom_tooltip = {
			text = "ztr_imf_scholarships_cooldown_tt"
			NOT = {
				has_variable = imf_scholarships_cooldown
			}
		}
		custom_tooltip = {
			text = "ztr_imf_scholarships_gp_relations_tt"
			any_country = {
				country_rank >= rank_value:great_power
				NOT = {
					this = root
				}
				relations:root >= relations_threshold:cordial
			}
		}
	}
	effect = {
		set_variable = {
			name = imf_scholarships_cooldown
			years = 3
		}
		add_modifier = {
			name = imf_foreign_scholarships_modifier
			years = 5
		}
	}
	ai_chance = {
		value = 1
		if = {
			limit = {
				literacy_rate < 0.6
			}
			add = 4
		}
		if = {
			limit = {
				get_country_gdp_per < 5
			}
			add = 2
		}
		if = {
			limit = {
				has_modifier = imf_technical_mission_modifier
			}
			add = 1
		}
		if = {
			limit = {
				scaled_debt > 0.25
			}
			multiply = 0
		}
	}
}
