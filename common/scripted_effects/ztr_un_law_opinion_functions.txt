# Here we have all the functions needed for UN
#1. ztr_un_random_subject_and_law_extraction -> Randomly selects a subject and a law for the UN to vote on
#2. ztr_un_set_possible_law_as_considered -> Sets the possible law as considered
#3. ztr_un_set_current_law_as_considered -> Sets the current law as considered
#4. ztr_un_set_vote_proposal_as_considered -> Sets the active vote proposal as considered
#5. ztr_un_set_current_vote_law_as_considered -> Sets the current law of the active vote subject as considered
#6. ztr_un_ai_try_start_possible_vote -> Tries to start a possible vote
#7. ztr_un_ai_take_neutral_vote_position -> Lets a neutral AI country pick support or against during a live vote
#8. ztr_un_value_law_prepare_country_context -> Prepares the country context for the law
#9. ztr_un_value_law -> Used by country to decide favor, neutral or against and by proposer if it wants to propose the law

ztr_un_ai_try_start_possible_vote = {
	# Fallback that after UN creation 100% doesn't do anything
	ztr_un_init_governance_laws = yes

	remove_variable = ztr_un_ai_current_law_value
	remove_variable = ztr_un_ai_possible_law_value
	remove_variable = ztr_un_subject_considered
	remove_variable = ztr_un_law_considered

	if = {
		limit = {
			ztr_un_country_can_propose_possible_law = yes
			ztr_un_is_not_applied_law = yes
		}

		# Current Law Check
		ztr_un_set_current_law_as_considered = yes
		ztr_un_value_law = yes
		set_variable = {
			name = ztr_un_ai_current_law_value
			value = var:ztr_un_law_value_total_considered
		}

		# Possible Law Check
		ztr_un_set_possible_law_as_considered = yes
		ztr_un_value_law = yes
		set_variable = {
			name = ztr_un_ai_possible_law_value
			value = var:ztr_un_law_value_total_considered
		}

		if = {
			limit = {
				var:ztr_un_ai_possible_law_value > var:ztr_un_ai_current_law_value
			}
			set_global_variable = {
				name = ztr_un_vote_subject
				value = global_var:ztr_un_subject_possible_proposal
			}
			set_global_variable = {
				name = ztr_un_vote_proposal
				value = global_var:ztr_un_law_possible_proposal
			}
			ztr_un_start_governance_vote = yes
			set_variable = ztr_un_vote_proposer
		}
	}

	remove_variable = ztr_un_ai_current_law_value
	remove_variable = ztr_un_ai_possible_law_value
	remove_variable = ztr_un_subject_considered
	remove_variable = ztr_un_law_considered
}

ztr_un_ai_take_neutral_vote_position = {
	ztr_un_init_governance_laws = yes

	remove_variable = ztr_un_ai_vote_current_law_value
	remove_variable = ztr_un_ai_vote_proposed_law_value
	remove_variable = ztr_un_ai_vote_support_margin
	remove_variable = ztr_un_ai_vote_against_margin
	remove_variable = ztr_un_ai_vote_threshold
	remove_variable = ztr_un_subject_considered
	remove_variable = ztr_un_law_considered

	if = {
		limit = {
			has_variable = ztr_un_vote_eligible
			NOT = {
				has_variable = ztr_un_vote_choice_support
			}
			NOT = {
				has_variable = ztr_un_vote_choice_against
			}
			has_global_variable = ztr_un_vote_subject
			has_global_variable = ztr_un_vote_proposal
			has_global_variable = ztr_un_vote_months_remaining
		}
		set_variable = {
			name = ztr_un_ai_vote_threshold
			value = global_var:ztr_un_vote_months_remaining
		}
		change_variable = {
			name = ztr_un_ai_vote_threshold
			multiply = 2
		}
		change_variable = {
			name = ztr_un_ai_vote_threshold
			add = 8
		}

		ztr_un_set_current_vote_law_as_considered = yes
		ztr_un_value_law = yes
		set_variable = {
			name = ztr_un_ai_vote_current_law_value
			value = var:ztr_un_law_value_total_considered
		}

		ztr_un_set_vote_proposal_as_considered = yes
		ztr_un_value_law = yes
		set_variable = {
			name = ztr_un_ai_vote_proposed_law_value
			value = var:ztr_un_law_value_total_considered
		}

		set_variable = {
			name = ztr_un_ai_vote_support_margin
			value = var:ztr_un_ai_vote_proposed_law_value
		}
		change_variable = {
			name = ztr_un_ai_vote_support_margin
			add = -var:ztr_un_ai_vote_current_law_value
		}

		set_variable = {
			name = ztr_un_ai_vote_against_margin
			value = var:ztr_un_ai_vote_current_law_value
		}
		change_variable = {
			name = ztr_un_ai_vote_against_margin
			add = -var:ztr_un_ai_vote_proposed_law_value
		}

		if = {
			limit = {
				var:ztr_un_ai_vote_support_margin > var:ztr_un_ai_vote_threshold
			}
			if = {
				limit = {
					has_variable = ztr_un_vote_choice_neutral
				}
				remove_variable = ztr_un_vote_choice_neutral
			}
			set_variable = ztr_un_vote_choice_support
		}
		else_if = {
			limit = {
				var:ztr_un_ai_vote_against_margin > var:ztr_un_ai_vote_threshold
			}
			if = {
				limit = {
					has_variable = ztr_un_vote_choice_neutral
				}
				remove_variable = ztr_un_vote_choice_neutral
			}
			set_variable = ztr_un_vote_choice_against
		}
	}

	remove_variable = ztr_un_ai_vote_current_law_value
	remove_variable = ztr_un_ai_vote_proposed_law_value
	remove_variable = ztr_un_ai_vote_support_margin
	remove_variable = ztr_un_ai_vote_against_margin
	remove_variable = ztr_un_ai_vote_threshold
	remove_variable = ztr_un_subject_considered
	remove_variable = ztr_un_law_considered
}
