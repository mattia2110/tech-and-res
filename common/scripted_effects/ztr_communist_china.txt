ztr_communist_china_cleanup = {
	remove_variable = ztr_ccp_has_inspiration
	remove_variable = ztr_ccp_front_active
	remove_variable = ztr_ccp_front_progress
	remove_variable = ztr_ccp_progress_contribution
	remove_variable = ztr_ccp_united_front
	remove_variable = ztr_ccp_armed_front
	if = {
		limit = {
			has_modifier = ztr_ccp_repression_campaign
		}
		remove_modifier = ztr_ccp_repression_campaign
	}
	if = {
		limit = {
			has_modifier = ztr_ccp_mass_line_campaign
		}
		remove_modifier = ztr_ccp_mass_line_campaign
	}
	if = {
		limit = {
			has_modifier = ztr_ccp_united_front_campaign
		}
		remove_modifier = ztr_ccp_united_front_campaign
	}
}

ztr_communist_china_clamp_overall_progress = {
	if = {
		limit = {
			global_var:ztr_ccp_overall_progress < 0
		}
		set_global_variable = {
			name = ztr_ccp_overall_progress
			value = 0
		}
	}
	if = {
		limit = {
			global_var:ztr_ccp_overall_progress > 100
		}
		set_global_variable = {
			name = ztr_ccp_overall_progress
			value = 100
		}
	}
}

ztr_communist_china_recalculate_overall_progress = {
	if = {
		limit = {
			has_global_variable = ztr_ccp_overall_progress
		}
		remove_global_variable = ztr_ccp_overall_progress
	}
	set_global_variable = {
		name = ztr_ccp_overall_progress
		value = 0
	}
	every_country = {
		limit = {
			has_variable = ztr_ccp_front_active
		}
		set_variable = {
			name = ztr_ccp_progress_contribution
			value = this.var:ztr_ccp_front_progress
		}
		change_variable = {
			name = ztr_ccp_progress_contribution
			divide = 5
		}
		change_global_variable = {
			name = ztr_ccp_overall_progress
			add = this.var:ztr_ccp_progress_contribution
		}
		remove_variable = ztr_ccp_progress_contribution
	}
	ztr_communist_china_clamp_overall_progress = yes
}

ztr_communist_china_prepare_scopes = {
	remove_variable = ztr_ccp_has_inspiration
	if = {
		limit = {
			any_country = {
				NOT = {
					this = root
				}
				country_rank >= rank_value:major_power
				ztr_is_communist = yes
			}
		}
		set_variable = ztr_ccp_has_inspiration
		random_country = {
			limit = {
				NOT = {
					this = root
				}
				country_rank >= rank_value:major_power
				ztr_is_communist = yes
			}
			save_scope_as = ztr_ccp_inspiration_country
		}
	}
}

ztr_communist_china_yearly_drift = {
	if = {
		limit = {
			has_variable = ztr_ccp_front_active
		}
		change_variable = {
			name = ztr_ccp_front_progress
			add = 4
		}
		if = {
			limit = {
				has_modifier = ztr_ccp_repression_campaign
			}
			change_variable = {
				name = ztr_ccp_front_progress
				subtract = 8
			}
		}
		if = {
			limit = {
				has_modifier = ztr_ccp_mass_line_campaign
			}
			change_variable = {
				name = ztr_ccp_front_progress
				add = 10
			}
		}
		if = {
			limit = {
				has_modifier = ztr_ccp_united_front_campaign
			}
			change_variable = {
				name = ztr_ccp_front_progress
				add = 8
			}
		}
		if = {
			limit = {
				has_variable = ztr_ccp_united_front
			}
			change_variable = {
				name = ztr_ccp_front_progress
				add = 5
			}
		}
		if = {
			limit = {
				has_variable = ztr_ccp_armed_front
			}
			change_variable = {
				name = ztr_ccp_front_progress
				add = 12
			}
		}
		if = {
			limit = {
				ztr_is_communist = yes
			}
			change_variable = {
				name = ztr_ccp_front_progress
				add = 6
			}
		}
		if = {
			limit = {
				is_subject = yes
			}
			change_variable = {
				name = ztr_ccp_front_progress
				add = 4
			}
		}
		if = {
			limit = {
				var:ztr_ccp_front_progress < 0
			}
			set_variable = {
				name = ztr_ccp_front_progress
				value = 0
			}
		}
		if = {
			limit = {
				var:ztr_ccp_front_progress > 100
			}
			set_variable = {
				name = ztr_ccp_front_progress
				value = 100
			}
		}
	}
}

ztr_communist_china_fire_yearly_event = {
	ztr_communist_china_prepare_scopes = yes
	if = {
		limit = {
			NOT = {
				has_variable = ztr_ccp_front_active
			}
		}
		trigger_event = {
			id = ztr_communist_china.1
			popup = yes
		}
	}
	#First country to obtain 60% progress in the front will trigger this event
	else_if = {
		limit = {
			global_var:ztr_ccp_overall_progress >= 50
			ztr_can_become_communist_chi_via_je = yes
		}
		trigger_event = {
			id = ztr_communist_china.4
			popup = yes
		}
	}
	#If the front is active but progress is low
	else_if = {
		limit = {
			OR = {
				var:ztr_ccp_front_progress < 40
				global_var:ztr_ccp_overall_progress < 20
			}
		}
		trigger_event = {
			id = ztr_communist_china.2
			popup = yes
		}
	}
	#If the front is active but progress is medium
	else_if = {
		limit = {
			OR = {
				var:ztr_ccp_front_progress < 60
				global_var:ztr_ccp_overall_progress < 40
			}
		}
		trigger_event = {
			id = ztr_communist_china.3
			popup = yes
		}
	}
}

ztr_communist_china_adopt_laws = {
	every_interest_group = {
		limit = {
			is_in_government = yes
		}
		remove_ruling_interest_group = yes
	}
	ig:ig_trade_unions ?= {
		add_ruling_interest_group = yes
	}
	ig:ig_rural_folk ?= {
		add_ruling_interest_group = yes
	}
	ig:ig_armed_forces ?= {
		add_ruling_interest_group = yes
	}
	ig:ig_trade_unions ?= {
		leader = {
			set_character_as_ruler = yes
		}
	}
	if = {
		limit = {
			NOT = {
				has_law_or_variant = law_type:law_council_republic
			}
		}
		activate_law = law_type:law_council_republic
	}
	activate_law = law_type:law_single_party_state
	activate_law = law_type:law_command_economy
	activate_law = law_type:law_collectivized_agriculture
	activate_law = law_type:law_state_atheism
	activate_law = law_type:law_appointed_bureaucrats
	activate_law = law_type:law_national_militia
	activate_law = law_type:law_secret_police
	activate_law = law_type:law_public_health_insurance
	activate_law = law_type:law_public_schools
}

ztr_nationalist_china_adopt_laws = {
	activate_law = law_type:law_presidential_republic
	activate_law = law_type:law_census_voting
	activate_law = law_type:law_professional_army
}

ztr_communist_china_create_nationalist_front = {
	if = {
		limit = {
			NOT = {
				exists = c:ZFC
			}
		}
		if = {
			limit = {
				any_country = {
					has_variable = ztr_ccp_nationalist_candidate
					count > 0
				}
			}
			ordered_country = {
				limit = {
					has_variable = ztr_ccp_nationalist_candidate
				}
				order_by = gdp
				position = 0
				check_range_bounds = no
				save_scope_as = ztr_ccp_nationalist_founder
			}
			scope:ztr_ccp_nationalist_founder = {
				change_tag = ZFC
				ztr_nationalist_china_adopt_laws = yes
			}
		}
	}
	every_country = {
		limit = {
			has_variable = ztr_ccp_nationalist_candidate
		}
		remove_variable = ztr_ccp_nationalist_candidate
	}
}

ztr_communist_china_create_front = {
	if = {
		limit = {
			NOT = {
				exists = c:CHI
			}
		}
		if = {
			limit = {
				NOT = {
					has_modifier = ztr_ccp_long_march_legacy
				}
			}
			add_modifier = {
				name = ztr_ccp_long_march_legacy
				years = 5
			}
		}
		change_tag = CHI
		ztr_communist_china_adopt_laws = yes
		set_global_variable = ztr_communist_china_created
		ztr_communist_china_cleanup = yes
	}
	else_if = {
		limit = {
			exists = c:CHI
			NOT = {
				is_player = yes
			}
		}
		c:CHI = {
			annex = root
		}
	}
	# Player will become communist CHI
	else_if = {
		limit = {
			is_player = yes
			NOT = {
				c:CHI = this
			}
		}
		custom_tooltip = {
			text = ztr_tooltip_can_become_communist_chi_via_je
			annex = c:CHI
			change_tag = CHI
			ztr_communist_china_adopt_laws = yes
			ztr_communist_china_cleanup = yes
		}
	}
}
