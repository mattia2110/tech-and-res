remove_imf_modifier = {
	remove_modifier = extreme_imf_help_modifier
	remove_modifier = big_imf_help_modifier
	remove_modifier = medium_imf_help_modifier
	remove_modifier = small_imf_help_modifier
	remove_modifier = tiny_imf_help_modifier
}

ztr_un_apply_member_institution_modifiers = {
	if = {
		limit = {
			has_global_variable = ztr_un_institution_imf
			global_var:ztr_un_institution_imf = 1
			NOT = {
				has_modifier = ztr_un_imf_member_modifier
			}
		}
		add_modifier = ztr_un_imf_member_modifier
	}
	if = {
		limit = {
			has_global_variable = ztr_un_institution_wto
			global_var:ztr_un_institution_wto = 1
			NOT = {
				has_modifier = ztr_wto_member_modifier
			}
		}
		add_modifier = ztr_wto_member_modifier
	}
	if = {
		limit = {
			has_global_variable = ztr_un_institution_icc
			global_var:ztr_un_institution_icc = 1
			NOT = {
				has_modifier = ztr_icc_member_modifier
			}
		}
		add_modifier = ztr_icc_member_modifier
	}
	if = {
		limit = {
			has_global_variable = ztr_un_institution_unesco
			global_var:ztr_un_institution_unesco = 1
			NOT = {
				has_modifier = ztr_unesco_member_modifier
			}
		}
		add_modifier = ztr_unesco_member_modifier
	}
}

ztr_un_remove_member_institution_modifiers = {
	if = {
		limit = {
			has_modifier = ztr_wto_member_modifier
		}
		remove_modifier = ztr_wto_member_modifier
	}
	if = {
		limit = {
			has_modifier = ztr_icc_member_modifier
		}
		remove_modifier = ztr_icc_member_modifier
	}
	if = {
		limit = {
			has_modifier = ztr_unesco_member_modifier
		}
		remove_modifier = ztr_unesco_member_modifier
	}
}
