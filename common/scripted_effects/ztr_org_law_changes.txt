join_un_law_changes = {
	if = {
		limit = {
			has_law = law_type:law_no_health_system
		}
		activate_law = law_type:law_charitable_health_system
	}
	# The United Nations will activate a restricted child labor law if child labor is allowed
	if = {
		limit = {
			has_law = law_type:law_restricted_child_labor
		}
		activate_law = law_type:law_compulsory_primary_school
	}
	if = {
		limit = {
			has_law = law_type:law_child_labor_allowed
		}
		activate_law = law_type:law_restricted_child_labor
	}
	# The United Nations will abolish outlawed dissent
	if = {
		limit = {
			OR = {
				has_law = law_type:law_outlawed_dissent
			}
		}
		activate_law = law_type:law_censorship
	}
	# The United Nations will adopt women's rights
	if = {
		limit = {
			OR = {
				has_law = law_type:law_no_womens_rights
			}
		}
		activate_law = law_type:law_women_own_property
	}
	# The United Nations will abolish racist laws if they are still present
	if = {
		limit = {
			OR = {
				has_law = law_type:law_ethnostate
				has_law = law_type:law_national_supremacy
			}
		}
		activate_law = law_type:law_racial_segregation
	}
	#TODO Modificare questa parte con la modifica forzata delle leggi attive per ogni membro dell'ONU
	# Apply active UN Human Rights resolutions to new members
	# Decolonization Stance
	if = {
		limit = {
			has_global_variable = ztr_un_hr_decolonization
			global_var:ztr_un_hr_decolonization = 2
			OR = {
				has_law = law_type:law_colonial_resettlement
				has_law = law_type:law_colonial_exploitation
				has_law = law_type:law_frontier_colonization
			}
		}
		activate_law = law_type:law_no_colonial_affairs
	}
	else_if = {
		limit = {
			has_global_variable = ztr_un_hr_decolonization
			global_var:ztr_un_hr_decolonization = 1
			OR = {
				has_law = law_type:law_colonial_resettlement
				has_law = law_type:law_colonial_exploitation
			}
		}
		activate_law = law_type:law_no_colonial_affairs
	}
	# Slavery Stance
	if = {
		limit = {
			has_global_variable = ztr_un_hr_slavery
			global_var:ztr_un_hr_slavery = 1
			OR = {
				has_law = law_type:law_colonial_slavery
				has_law = law_type:law_debt_slavery
				has_law = law_type:law_slave_trade
				has_law = law_type:law_legacy_slavery
			}
		}
		activate_law = law_type:law_slavery_banned
	}
}

join_eu_law_changes = {
	# Universal suffrage
	if = {
		limit = {
			NOT = { has_law = law_type:law_universal_suffrage }
		}
		activate_law = law_type:law_universal_suffrage
	}
	# Parliamentary republic
	if = {
		limit = {
			OR = {
				has_law = law_type:law_corporate_state
				has_law = law_type:law_theocracy
			}
		}
		activate_law = law_type:law_parliamentary_republic
	}
	# Cultural exclusion
	if = {
		limit = {
			NOR = {
				has_law = law_type:law_multicultural
				has_law = law_type:law_cultural_exclusion
			}
		}
		activate_law = law_type:law_cultural_exclusion
	}
	# Compulsory primary school
	if = {
		limit = {
			NOT = {
				has_law = law_type:law_compulsory_primary_school
			}
		}
		activate_law = law_type:law_compulsory_primary_school
	}
	# Women's suffrage
	if = {
		limit = {
			NOT = {
				has_law = law_type:law_womens_suffrage
			}
		}
		activate_law = law_type:law_womens_suffrage
	}
	# Interventionism
	if = {
		limit = {
			NOR = {
				has_law = law_type:law_interventionism
				has_law = law_type:law_laissez_faire
			}
		}
		activate_law = law_type:law_interventionism
	}
	# Protectionism
	if = {
		limit = {
			NOR = {
				has_law = law_type:law_mercantilism
				has_law = law_type:law_isolationism
			}
		}
		activate_law = law_type:law_protectionism
	}
	# Protected speech
	if = {
		limit = {
			NOT = {
				has_law = law_type:law_protected_speech
			}
		}
		activate_law = law_type:law_protected_speech
	}
}
