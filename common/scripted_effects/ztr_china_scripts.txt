ztr_china_update_corporate_bar = {
	if = {
		limit = {
			ig:ig_industrialists = {
				is_powerful = yes
			}
		}
		change_variable = {
			name = ztr_china_corporate_component_var
			add = 1
		}
	}
	if = {
		limit = {
			ig:ig_industrialists = {
				is_in_government = yes
			}
		}
		change_variable = {
			name = ztr_china_corporate_component_var
			add = 0.5
		}
	}
	if = {
		limit = {
			has_law = law_type:law_free_trade
		}
		change_variable = {
			name = ztr_china_corporate_component_var
			add = 0.5
		}
	}
	if = {
		limit = {
			has_law = law_type:law_laissez_faire
		}
		change_variable = {
			name = ztr_china_corporate_component_var
			add = 0.5
		}
	}
	if = {
		limit = {
			ztr_gdp_per_capita >= 2
		}
		change_variable = {
			name = ztr_china_corporate_component_var
			add = 0.5
		}
	}
	if = {
		limit = {
			ztr_gdp_per_capita >= 4
		}
		change_variable = {
			name = ztr_china_corporate_component_var
			add = 0.5
		}
	}
	if = {
		limit = {
			has_technology_researched = stock_exchange
		}
		change_variable = {
			name = ztr_china_corporate_component_var
			add = 0.5
		}
	}
	if = {
		limit = {
			ig:ig_landowners = {
				is_powerful = yes
			}
		}
		change_variable = {
			name = ztr_china_corporate_component_var
			subtract = 0.5
		}
	}
	if = {
		limit = {
			OR = {
				has_law = law_type:law_isolationism
				has_law = law_type:law_canton_system
			}
		}
		change_variable = {
			name = ztr_china_corporate_component_var
			subtract = 0.5
		}
	}
	if = {
		limit = {
			any_scope_state = {
				turmoil > 0.25
			}
		}
		change_variable = {
			name = ztr_china_corporate_component_var
			subtract = 0.5
		}
	}
}

ztr_china_final_renaissance_update = {
	# Aggregate components into variables
	change_variable = {
		name = ztr_china_corporate_var
		add = var:ztr_china_corporate_component_var
	}
	change_variable = {
		name = ztr_china_federation_var
		add = var:ztr_china_federation_component_var
	}
	change_variable = {
		name = ztr_china_technoconfucian_var
		add = var:ztr_china_technoconfucian_component_var
	}
	# Keep variables at minimum 0
	if = {
		limit = {
			var:ztr_china_corporate_var < 0
		}
		set_variable = {
			name = ztr_china_corporate_var
			value = 0
		}
	}
	if = {
		limit = {
			var:ztr_china_federation_var < 0
		}
		set_variable = {
			name = ztr_china_federation_var
			value = 0
		}
	}
	if = {
		limit = {
			var:ztr_china_technoconfucian_var < 0
		}
		set_variable = {
			name = ztr_china_technoconfucian_var
			value = 0
		}
	}
	# Update UI bars
	scope:journal_entry = {
		set_bar_progress = {
			value = root.var:ztr_china_corporate_var
			name = ztr_china_corporate_progress_bar
		}
		set_bar_progress = {
			value = root.var:ztr_china_federation_var
			name = ztr_china_federation_progress_bar
		}
		set_bar_progress = {
			value = root.var:ztr_china_technoconfucian_var
			name = ztr_china_technoconfucian_progress_bar
		}
	}
	# Clear temporary component variables
	set_variable = {
		name = ztr_china_corporate_component_var
		value = 0
	}
	set_variable = {
		name = ztr_china_federation_component_var
		value = 0
	}
	set_variable = {
		name = ztr_china_technoconfucian_component_var
		value = 0
	}
}

ztr_communist_china_clamp_overall_progress = {
	if = {
		limit = {
			global_var:ztr_ccp_overall_progress < 0
		}
		set_global_variable = {
			name = ztr_ccp_overall_progress
			value = 0
		}
	}
	if = {
		limit = {
			global_var:ztr_ccp_overall_progress > 100
		}
		set_global_variable = {
			name = ztr_ccp_overall_progress
			value = 100
		}
	}
}

ztr_communist_china_recalculate_overall_progress = {
	if = {
		limit = {
			has_global_variable = ztr_ccp_overall_progress
		}
		remove_global_variable = ztr_ccp_overall_progress
	}
	set_global_variable = {
		name = ztr_ccp_overall_progress
		value = 0
	}
	every_country = {
		limit = {
			has_variable = ztr_ccp_front_active
		}
		set_variable = {
			name = ztr_ccp_progress_contribution
			value = this.var:ztr_ccp_front_progress
		}
		change_variable = {
			name = ztr_ccp_progress_contribution
			divide = 5
		}
		change_global_variable = {
			name = ztr_ccp_overall_progress
			add = this.var:ztr_ccp_progress_contribution
		}
		remove_variable = ztr_ccp_progress_contribution
	}
	ztr_communist_china_clamp_overall_progress = yes
}

ztr_communist_china_prepare_scopes = {
	remove_variable = ztr_ccp_has_inspiration
	if = {
		limit = {
			any_country = {
				NOT = {
					this = root
				}
				country_rank >= rank_value:major_power
				ztr_is_communist = yes
			}
		}
		set_variable = ztr_ccp_has_inspiration
		random_country = {
			limit = {
				NOT = {
					this = root
				}
				country_rank >= rank_value:major_power
				ztr_is_communist = yes
			}
			save_scope_as = ztr_ccp_inspiration_country
		}
	}
}

ztr_communist_china_fire_yearly_event = {
	ztr_communist_china_prepare_scopes = yes
	if = {
		limit = {
			NOT = {
				has_variable = ztr_ccp_front_active
			}
		}
		trigger_event = {
			id = ztr_communist_china.1
			popup = yes
		}
	}
	#First country to obtain 60% progress in the front will trigger this event
	else_if = {
		limit = {
			global_var:ztr_ccp_overall_progress >= 50
			ztr_can_become_communist_chi_via_je = yes
		}
		trigger_event = {
			id = ztr_communist_china.4
			popup = yes
		}
	}
	#If the front is active but progress is low
	else_if = {
		limit = {
			OR = {
				var:ztr_ccp_front_progress < 40
				global_var:ztr_ccp_overall_progress < 20
			}
		}
		trigger_event = {
			id = ztr_communist_china.2
			popup = yes
		}
	}
	#If the front is active but progress is medium
	else_if = {
		limit = {
			OR = {
				var:ztr_ccp_front_progress < 60
				global_var:ztr_ccp_overall_progress < 40
			}
		}
		trigger_event = {
			id = ztr_communist_china.3
			popup = yes
		}
	}
}

		if = {
			limit = {

					has_variable = ztr_ccp_nationalist_candidate

				}
			}
			ordered_country = {
				limit = {
					has_variable = ztr_ccp_nationalist_candidate
				}
				order_by = gdp
				position = 0
				check_range_bounds = no
				save_scope_as = ztr_ccp_nationalist_founder
			}
			scope:ztr_ccp_nationalist_founder = {
				change_tag = ZFC
				ztr_nationalist_china_adopt_laws_and_loyalist = yes
			}
		}
	}

			has_variable = ztr_ccp_nationalist_candidate

	}

		ztr_communist_china_cleanup = yes

		limit = {
			exists = c:CHI
			NOT = {
				is_player = yes
			}
		}
		c:CHI = {
			annex = root
		}
	}
	# Player will become communist CHI
	else_if = {
		limit = {
			is_player = yes
			NOT = {
				c:CHI = this
			}
		}

			text = ztr_tooltip_can_become_communist_chi_via_je

			change_tag = CHI
			ztr_communist_china_adopt_laws_and_loyalist = yes

		}
	}
