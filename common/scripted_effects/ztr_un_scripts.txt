ztr_un_vote_clear_results = {
	remove_global_variable = ztr_un_vote_support_count
	remove_global_variable = ztr_un_vote_neutral_count
	remove_global_variable = ztr_un_vote_against_count
	remove_global_variable = ztr_un_vote_total_count
	remove_global_variable = ztr_un_vote_support_percent
	remove_global_variable = ztr_un_vote_neutral_percent
	remove_global_variable = ztr_un_vote_against_percent
}

join_un_law_changes = {
	if = {
		limit = {
			has_law_or_variant = law_type:law_no_health_system
		}
		activate_law = law_type:law_charitable_health_system
	}
	# The United Nations will abolish outlawed dissent
	if = {
		limit = {
			OR = {
				has_law_or_variant = law_type:law_outlawed_dissent
			}
		}
		activate_law = law_type:law_censorship
	}
	# The United Nations will abolish racist laws if they are still present
	if = {
		limit = {
			OR = {
				has_law_or_variant = law_type:law_ethnostate
				has_law_or_variant = law_type:law_national_supremacy
			}
		}
		activate_law = law_type:law_racial_segregation
	}
	#TODO Modificare questa parte con la modifica forzata delle leggi attive per ogni membro dell'ONU
	# Apply active UN Human Rights resolutions to new members
	# Decolonization Stance
	# Encouraged
	if = {
		limit = {
			has_global_variable = ztr_un_hr_decolonization
			global_var:ztr_un_hr_decolonization = 2
			OR = {
				has_law_or_variant = law_type:law_colonial_resettlement
				has_law_or_variant = law_type:law_colonial_exploitation
				has_law_or_variant = law_type:law_frontier_colonization
			}
		}
		activate_law = law_type:law_no_colonial_affairs
	}
	# Conservative
	else_if = {
		limit = {
			has_global_variable = ztr_un_hr_decolonization
			global_var:ztr_un_hr_decolonization = 1
			OR = {
				has_law_or_variant = law_type:law_colonial_resettlement
				has_law_or_variant = law_type:law_colonial_exploitation
			}
		}
		activate_law = law_type:law_no_colonial_affairs
	}
	# Slavery Stance
	if = {
		limit = {
			has_global_variable = ztr_un_hr_slavery
			global_var:ztr_un_hr_slavery = 1
			OR = {
				has_law_or_variant = law_type:law_colonial_slavery
				has_law_or_variant = law_type:law_debt_slavery
				has_law_or_variant = law_type:law_slave_trade
				has_law_or_variant = law_type:law_legacy_slavery
			}
		}
		activate_law = law_type:law_slavery_banned
		set_variable = {
			name = ztr_recently_banned_slavery
			years = 10
		}
	}
	#Children's Rights
	if = {
		limit = {
			has_global_variable = ztr_un_hr_children_rights
			global_var:ztr_un_hr_children_rights = 1
			has_law_or_variant = law_type:law_child_labor_allowed
		}
		activate_law = law_type:law_restricted_child_labor
	}
	else_if = {
		limit = {
			has_global_variable = ztr_un_hr_children_rights
			global_var:ztr_un_hr_children_rights = 2
			OR = {
				has_law_or_variant = law_type:law_child_labor_allowed
				has_law_or_variant = law_type:law_restricted_child_labor
			}
		}
		activate_law = law_type:law_compulsory_primary_school
	}
	#Women's rights
	if = {
		limit = {
			has_global_variable = ztr_un_hr_women_rights
			global_var:ztr_un_hr_women_rights = 1
			OR = {
				has_law_or_variant = law_type:law_no_womens_rights
				has_law_or_variant = law_type:law_women_own_property
			}
		}
		activate_law = law_type:law_women_in_the_workplace
	}
	ztr_un_apply_member_institution_modifiers = yes
}

ztr_un_vote_recalculate_results = {
	ztr_un_vote_clear_results = yes
	set_global_variable = {
		name = ztr_un_vote_support_count
		value = 0
	}
	set_global_variable = {
		name = ztr_un_vote_neutral_count
		value = 0
	}
	set_global_variable = {
		name = ztr_un_vote_against_count
		value = 0
	}
	set_global_variable = {
		name = ztr_un_vote_total_count
		value = 0
	}
	every_scope_je_involved = {
		# Use vote weight if set (Weighted by Prestige model), otherwise default to 1
		if = {
			limit = {
				has_variable = ztr_un_vote_weight
			}
			change_global_variable = {
				name = ztr_un_vote_total_count
				add = var:ztr_un_vote_weight
			}
			if = {
				limit = {
					has_variable = ztr_un_vote_choice_support
				}
				change_global_variable = {
					name = ztr_un_vote_support_count
					add = var:ztr_un_vote_weight
				}
			}
			else_if = {
				limit = {
					has_variable = ztr_un_vote_choice_against
				}
				change_global_variable = {
					name = ztr_un_vote_against_count
					add = var:ztr_un_vote_weight
				}
			}
			else = {
				change_global_variable = {
					name = ztr_un_vote_neutral_count
					add = var:ztr_un_vote_weight
				}
			}
		}
		else = {
			change_global_variable = {
				name = ztr_un_vote_total_count
				add = 1
			}
			if = {
				limit = {
					has_variable = ztr_un_vote_choice_support
				}
				change_global_variable = {
					name = ztr_un_vote_support_count
					add = 1
				}
			}
			else_if = {
				limit = {
					has_variable = ztr_un_vote_choice_against
				}
				change_global_variable = {
					name = ztr_un_vote_against_count
					add = 1
				}
			}
			else = {
				change_global_variable = {
					name = ztr_un_vote_neutral_count
					add = 1
				}
			}
		}
	}
	set_global_variable = {
		name = ztr_un_vote_support_percent
		value = 0
	}
	set_global_variable = {
		name = ztr_un_vote_neutral_percent
		value = 0
	}
	set_global_variable = {
		name = ztr_un_vote_against_percent
		value = 0
	}
	if = {
		limit = {
			global_var:ztr_un_vote_total_count > 0
		}
		set_global_variable = {
			name = ztr_un_vote_support_percent
			value = global_var:ztr_un_vote_support_count
		}
		change_global_variable = {
			name = ztr_un_vote_support_percent
			divide = global_var:ztr_un_vote_total_count
		}
		change_global_variable = {
			name = ztr_un_vote_support_percent
			multiply = 100
		}
		set_global_variable = {
			name = ztr_un_vote_neutral_percent
			value = global_var:ztr_un_vote_neutral_count
		}
		change_global_variable = {
			name = ztr_un_vote_neutral_percent
			divide = global_var:ztr_un_vote_total_count
		}
		change_global_variable = {
			name = ztr_un_vote_neutral_percent
			multiply = 100
		}
		set_global_variable = {
			name = ztr_un_vote_against_percent
			value = global_var:ztr_un_vote_against_count
		}
		change_global_variable = {
			name = ztr_un_vote_against_percent
			divide = global_var:ztr_un_vote_total_count
		}
		change_global_variable = {
			name = ztr_un_vote_against_percent
			multiply = 100
		}
	}
}

# Start a UN vote on a governance proposal
# Expects ztr_un_vote_subject and ztr_un_vote_proposal to be set before calling
ztr_un_start_governance_vote = {
	set_global_variable = ztr_un_vote_in_progress
	set_global_variable = {
		name = ztr_un_vote_months_remaining
		value = 12
	}
	every_country = {
		limit = {
			has_variable = ztr_un_vote_eligible
		}
		remove_variable = ztr_un_vote_eligible
		remove_variable = ztr_un_vote_choice_support
		remove_variable = ztr_un_vote_choice_neutral
		remove_variable = ztr_un_vote_choice_against
		remove_variable = ztr_un_vote_weight
		remove_variable = ztr_un_vote_proposer
	}
	# Model 1: One Country One Vote - all UN members
	if = {
		limit = {
			global_var:ztr_un_law_voting_model = 1
		}
		every_country = {
			limit = {
				OR = {
					has_modifier = ztr_onu_member
					has_modifier = ztr_onu_permanent_member
				}
			}
			set_variable = ztr_un_vote_eligible
			set_variable = {
				name = ztr_un_vote_weight
				value = 1
			}
			trigger_event = {
				id = ztr_un_event.10
				days = 30
			}
		}
	}
	# Model 2: Weighted by Prestige - all UN members, weight by rank
	else_if = {
		limit = {
			global_var:ztr_un_law_voting_model = 2
		}
		every_country = {
			limit = {
				OR = {
					has_modifier = ztr_onu_member
					has_modifier = ztr_onu_permanent_member
				}
			}
			set_variable = ztr_un_vote_eligible
			# Great powers get 25 votes, major powers 10, minor powers 5, others 1
			if = {
				limit = {
					country_rank = rank_value:great_power
				}
				set_variable = {
					name = ztr_un_vote_weight
					value = 25
				}
			}
			else_if = {
				limit = {
					country_rank = rank_value:major_power
				}
				set_variable = {
					name = ztr_un_vote_weight
					value = 10
				}
			}
			else_if = {
				limit = {
					country_rank = rank_value:minor_power
				}
				set_variable = {
					name = ztr_un_vote_weight
					value = 5
				}
			}
			else = {
				set_variable = {
					name = ztr_un_vote_weight
					value = 1
				}
			}
			trigger_event = {
				id = ztr_un_event.10
				days = 30
			}
		}
	}
	# Model 3: Security Council Only - only permanent members
	else_if = {
		limit = {
			global_var:ztr_un_law_voting_model = 3
		}
		every_country = {
			limit = {
				has_modifier = ztr_onu_permanent_member
			}
			set_variable = ztr_un_vote_eligible
			set_variable = {
				name = ztr_un_vote_weight
				value = 1
			}
			trigger_event = {
				id = ztr_un_event.10
				days = 30
			}
		}
	}
	# Model 4: Superpowers Only - top 2 ranked countries in UN
	else_if = {
		limit = {
			global_var:ztr_un_law_voting_model = 4
		}
		ordered_country = {
			limit = {
				OR = {
					has_modifier = ztr_onu_member
					has_modifier = ztr_onu_permanent_member
				}
			}
			order_by = prestige
			max = 2
			check_range_bounds = no
			set_variable = ztr_un_vote_eligible
			set_variable = {
				name = ztr_un_vote_weight
				value = 1
			}
			trigger_event = {
				id = ztr_un_event.10
				days = 30
			}
		}
	}
}

# Apply the voted governance change (called from je_un_votations on_timeout when vote passes)
ztr_un_apply_governance_vote = {
	# Subject 1: Voting Model
	if = {
		limit = {
			has_global_variable = ztr_un_vote_subject
			global_var:ztr_un_vote_subject = 1
		}
		remove_global_variable = ztr_un_law_voting_model
		set_global_variable = {
			name = ztr_un_law_voting_model
			value = global_var:ztr_un_vote_proposal
		}
	}
	# Subject 2: Veto Right
	if = {
		limit = {
			has_global_variable = ztr_un_vote_subject
			global_var:ztr_un_vote_subject = 2
		}
		remove_global_variable = ztr_un_law_veto_right
		set_global_variable = {
			name = ztr_un_law_veto_right
			value = global_var:ztr_un_vote_proposal
		}
	}
	# Subject 3: Member Selection - paradigm shift
	if = {
		limit = {
			has_global_variable = ztr_un_vote_subject
			global_var:ztr_un_vote_subject = 3
		}
		remove_global_variable = ztr_un_law_member_selection
		set_global_variable = {
			name = ztr_un_law_member_selection
			value = global_var:ztr_un_vote_proposal
		}
		# Strip all current permanent members to regular members
		every_country = {
			limit = {
				has_modifier = ztr_onu_permanent_member
			}
			remove_modifier = ztr_onu_permanent_member
			add_modifier = ztr_onu_member
		}
		# Select new permanent members based on the new model
		ztr_un_select_new_permanent_members = yes
	}
	# Subject 4: Human Rights - Decolonization
	if = {
		limit = {
			has_global_variable = ztr_un_vote_subject
			global_var:ztr_un_vote_subject = 4
		}
		if = {
			limit = {
				has_global_variable = ztr_un_hr_decolonization
			}
			remove_global_variable = ztr_un_hr_decolonization
		}
		set_global_variable = {
			name = ztr_un_hr_decolonization
			value = global_var:ztr_un_vote_proposal
		}
		#Apply effects based on the stance
		# Decolonization Stance
		# Conservative Decolonization
		if = {
			limit = {
				global_var:ztr_un_hr_decolonization = 1
			}
			every_country = {
				limit = {
					ztr_is_un_member = yes
					OR = {
						has_law_or_variant = law_type:law_colonial_resettlement
						has_law_or_variant = law_type:law_colonial_exploitation
					}
				}
				activate_law = law_type:law_no_colonial_affairs
			}
		}
		# Encouraged Decolonization
		else_if = {
			limit = {
				global_var:ztr_un_hr_decolonization = 2
			}
			#TODO Pensare se dover aggiungere altro
			every_country = {
				limit = {
					ztr_is_un_member = yes
					OR = {
						has_law_or_variant = law_type:law_colonial_resettlement
						has_law_or_variant = law_type:law_colonial_exploitation
						has_law_or_variant = law_type:law_frontier_colonization
					}
				}
				activate_law = law_type:law_no_colonial_affairs
			}
		}
	}
	# Subject 5: Human Rights - Slavery
	if = {
		limit = {
			has_global_variable = ztr_un_vote_subject
			global_var:ztr_un_vote_subject = 5
		}
		if = {
			limit = {
				has_global_variable = ztr_un_hr_slavery
			}
			remove_global_variable = ztr_un_hr_slavery
		}
		set_global_variable = {
			name = ztr_un_hr_slavery
			value = global_var:ztr_un_vote_proposal
		}
		#Apply effects based on the stance
		# Illegal: add modifier to all UN members that disables slavery laws
		if = {
			limit = {
				global_var:ztr_un_hr_slavery = 1
			}
			every_country = {
				limit = {
					ztr_is_un_member = yes
					OR = {
						has_law_or_variant = law_type:law_colonial_slavery
						has_law_or_variant = law_type:law_debt_slavery
						has_law_or_variant = law_type:law_slave_trade
						has_law_or_variant = law_type:law_legacy_slavery
					}
				}
				activate_law = law_type:law_slavery_banned
				set_variable = {
					name = ztr_recently_banned_slavery
					years = 10
				}
			}
		}
	}
	# Subject 6: Human Rights - Children's Rights
	if = {
		limit = {
			has_global_variable = ztr_un_vote_subject
			global_var:ztr_un_vote_subject = 6
		}
		if = {
			limit = {
				has_global_variable = ztr_un_hr_children_rights
			}
			remove_global_variable = ztr_un_hr_children_rights
		}
		set_global_variable = {
			name = ztr_un_hr_children_rights
			value = global_var:ztr_un_vote_proposal
		}
		# Law enforcement is handled by the dedicated law scripts.
		# Children's rights - restricted child labor
		if = {
			limit = {
				global_var:ztr_un_hr_children_rights = 1
			}
			every_country = {
				limit = {
					ztr_is_un_member = yes
					has_law_or_variant = law_type:law_child_labor_allowed
				}
				activate_law = law_type:law_restricted_child_labor
			}
		}
		# Children's rights - compulsory primary education
		else_if = {
			limit = {
				global_var:ztr_un_hr_children_rights = 2
			}
			every_country = {
				limit = {
					ztr_is_un_member = yes
					OR = {
						has_law_or_variant = law_type:law_child_labor_allowed
						has_law_or_variant = law_type:law_restricted_child_labor
					}
				}
				activate_law = law_type:law_compulsory_primary_school
			}
		}
	}
	# Subject 7: Human Rights - Women's Rights
	if = {
		limit = {
			has_global_variable = ztr_un_vote_subject
			global_var:ztr_un_vote_subject = 7
		}
		if = {
			limit = {
				has_global_variable = ztr_un_hr_women_rights
			}
			remove_global_variable = ztr_un_hr_women_rights
		}
		set_global_variable = {
			name = ztr_un_hr_women_rights
			value = global_var:ztr_un_vote_proposal
		}
		# Law enforcement is handled by the dedicated law scripts.
		# Women's rights - gender equality
		if = {
			limit = {
				global_var:ztr_un_hr_women_rights = 1
			}
			every_country = {
				limit = {
					ztr_is_un_member = yes
					OR = {
						has_law_or_variant = law_type:law_no_womens_rights
						has_law_or_variant = law_type:law_women_own_property
					}
				}
				activate_law = law_type:law_women_in_the_workplace
			}
		}
	}
	# Subject 8: IMF (+ World Bank)
	if = {
		limit = {
			has_global_variable = ztr_un_vote_subject
			global_var:ztr_un_vote_subject = 8
		}
		if = {
			limit = {
				has_global_variable = ztr_un_institution_imf
			}
			remove_global_variable = ztr_un_institution_imf
		}
		set_global_variable = {
			name = ztr_un_institution_imf
			value = global_var:ztr_un_vote_proposal
		}
		every_country = {
			limit = {
				has_variable = ztr_un_vote_proposer
			}
			#TODO: Tenere a mente per estendere evento finale di legge UN ad altre elezioni
			trigger_event = ztr_imf_event.1
			remove_variable = ztr_un_vote_proposer
		}
	}
	# Subject 9: WTO
	if = {
		limit = {
			has_global_variable = ztr_un_vote_subject
			global_var:ztr_un_vote_subject = 9
		}
		if = {
			limit = {
				has_global_variable = ztr_un_institution_wto
			}
			remove_global_variable = ztr_un_institution_wto
		}
		set_global_variable = {
			name = ztr_un_institution_wto
			value = global_var:ztr_un_vote_proposal
		}
		every_country = {
			limit = {
				ztr_is_un_member = yes
			}
			ztr_un_apply_member_institution_modifiers = yes
		}
	}
	# Subject 10: International Criminal Court
	if = {
		limit = {
			has_global_variable = ztr_un_vote_subject
			global_var:ztr_un_vote_subject = 10
		}
		if = {
			limit = {
				has_global_variable = ztr_un_institution_icc
			}
			remove_global_variable = ztr_un_institution_icc
		}
		set_global_variable = {
			name = ztr_un_institution_icc
			value = global_var:ztr_un_vote_proposal
		}
		every_country = {
			limit = {
				ztr_is_un_member = yes
			}
			ztr_un_apply_member_institution_modifiers = yes
		}
	}
	# Subject 11: UNESCO
	if = {
		limit = {
			has_global_variable = ztr_un_vote_subject
			global_var:ztr_un_vote_subject = 11
		}
		if = {
			limit = {
				has_global_variable = ztr_un_institution_unesco
			}
			remove_global_variable = ztr_un_institution_unesco
		}
		set_global_variable = {
			name = ztr_un_institution_unesco
			value = global_var:ztr_un_vote_proposal
		}
		every_country = {
			limit = {
				ztr_is_un_member = yes
			}
			ztr_un_apply_member_institution_modifiers = yes
		}
	}
	# Clean up vote subject/proposal variables
	remove_global_variable = ztr_un_vote_subject
	remove_global_variable = ztr_un_vote_proposal
}

# Select new permanent members based on the current member selection law
ztr_un_select_new_permanent_members = {
	# Model 1: By Prestige - top 3 by prestige
	if = {
		limit = {
			global_var:ztr_un_law_member_selection = 1
		}
		ordered_country = {
			limit = {
				has_modifier = ztr_onu_member
			}
			order_by = prestige
			max = 3
			check_range_bounds = no
			remove_modifier = ztr_onu_member
			add_modifier = ztr_onu_permanent_member
			trigger_event = {
				id = ztr_un_event.25
				days = 7
			}
		}
	}
	# Model 2: Geographic Representation - the strongest of each major region
	else_if = {
		limit = {
			global_var:ztr_un_law_member_selection = 2
		}
		# Europe
		ordered_country = {
			limit = {
				has_modifier = ztr_onu_member
				is_in_geographic_region = geographic_region_europe
			}
			order_by = prestige
			max = 1
			check_range_bounds = no
			remove_modifier = ztr_onu_member
			add_modifier = ztr_onu_permanent_member
			trigger_event = {
				id = ztr_un_event.25
				days = 7
			}
		}
		# Americas
		ordered_country = {
			limit = {
				has_modifier = ztr_onu_member
				is_in_geographic_region = geographic_region_americas
			}
			order_by = prestige
			max = 1
			check_range_bounds = no
			remove_modifier = ztr_onu_member
			add_modifier = ztr_onu_permanent_member
			trigger_event = {
				id = ztr_un_event.25
				days = 7
			}
		}
		# East Asia / Southeast Asia / Oceania
		ordered_country = {
			limit = {
				has_modifier = ztr_onu_member
				OR = {
					is_in_geographic_region = geographic_region_east_asia
					is_in_geographic_region = geographic_region_south_east_asia
					is_in_geographic_region = geographic_region_oceania
				}
			}
			order_by = prestige
			max = 1
			check_range_bounds = no
			remove_modifier = ztr_onu_member
			add_modifier = ztr_onu_permanent_member
			trigger_event = {
				id = ztr_un_event.25
				days = 7
			}
		}
		# Middle East / Central Asia / India
		ordered_country = {
			limit = {
				has_modifier = ztr_onu_member
				OR = {
					is_in_geographic_region = geographic_region_middle_east
					is_in_geographic_region = geographic_region_central_asia
					is_in_geographic_region = geographic_region_india
				}
			}
			order_by = prestige
			max = 1
			check_range_bounds = no
			remove_modifier = ztr_onu_member
			add_modifier = ztr_onu_permanent_member
			trigger_event = {
				id = ztr_un_event.25
				days = 7
			}
		}
		# Africa
		ordered_country = {
			limit = {
				has_modifier = ztr_onu_member
				is_in_geographic_region = geographic_region_africa
			}
			order_by = prestige
			max = 1
			check_range_bounds = no
			remove_modifier = ztr_onu_member
			add_modifier = ztr_onu_permanent_member
			trigger_event = {
				id = ztr_un_event.25
				days = 7
			}
		}
	}
	# Model 3: By Military Power - top 3 by military strength
	else_if = {
		limit = {
			global_var:ztr_un_law_member_selection = 3
		}
		ordered_country = {
			limit = {
				has_modifier = ztr_onu_member
			}
			order_by = {
				value = army_power_projection
				add = naval_power_projection
			}
			max = 3
			check_range_bounds = no
			remove_modifier = ztr_onu_member
			add_modifier = ztr_onu_permanent_member
			trigger_event = {
				id = ztr_un_event.25
				days = 7
			}
		}
	}
	# Model 4: Strongest Country - top 1
	else_if = {
		limit = {
			global_var:ztr_un_law_member_selection = 4
		}
		random_country = {
			limit = {
				global_country_ranking = 1
			}
			remove_modifier = ztr_onu_member
			add_modifier = ztr_onu_permanent_member
			trigger_event = {
				id = ztr_un_event.25
				days = 7
			}
		}
	}
}
