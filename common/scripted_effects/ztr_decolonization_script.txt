british_raj_consolidation_effect = {
	# At first Great Britain will give all of its Indian states to the British Raj
	if = {
		limit = {
			c:BIC = {
				overlord = root
			}
		}
		root = {
			every_scope_state = {
				limit = {
					is_in_geographic_region = geographic_region_india
				}
				set_state_owner = c:BIC
			}
		}
	}
	# Then the British Raj will annex all of the Rajah states
	c:BIC = {
		every_subject_or_below = {
			limit = {
				is_in_geographic_region = geographic_region_india
			}
			every_scope_state = {
				limit = {
					is_in_geographic_region = geographic_region_india
				}
				set_state_owner = c:BIC
			}
		}
	}
}

give_claims_on_india = {
	every_state = {
		limit = {
			NOT = { owner = ROOT }
			is_in_geographic_region = geographic_region_india
			NOT = { state_region = s:STATE_CEYLON }
		}
		state_region = { add_claim = ROOT }
	}
}

calculate_infamy_decolonization_subject_event = {
	scope:subject_scope = {
		every_scope_state = { 
			root = { change_variable = { name = decolonization_subject_event_var add = 1 }}
		}
	}
	change_variable = {
		name = decolonization_subject_event_var
		multiply = 2
	}
	if = {
		limit = {
			var:decolonization_subject_event_var > 15
		}
		remove_variable = decolonization_subject_event_var
		set_variable = {
			name = decolonization_subject_event_var
			value = 15
		}
	}
}
