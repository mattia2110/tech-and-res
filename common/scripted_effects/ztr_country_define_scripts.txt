ztr_define_kenya = {
	cu:kikuyu = {
		save_scope_as = ztr_minor_primary_culture
	}
	ztr_state_finder_script = yes
	if = {
		limit = { NOT = { exists = c:ZKN } }
		create_country = {
			tag    = ZKN
			origin = this
			state  = SCOPE:ztr_state_finder
			on_created = {
				set_country_type = recognized
				activate_law     = law_type:law_parliamentary_republic
				activate_law     = law_type:law_census_voting
				activate_law     = law_type:law_no_schools
			}
		}
	}
	c:ZKN = {
		if = {
			limit = { religion = rel:animist }
			african_religion_setting = yes
		}
	}
	every_scope_state = {
		limit = {
			NOT = { is_homeland_of_country_cultures = root }
			state_region = {
				OR = {
					region = sr:region_zanj
					region = sr:region_ethiopia
				}
				is_homeland = scope:ztr_minor_primary_culture
			}
		}
		set_state_owner = c:ZKN
	}
}
