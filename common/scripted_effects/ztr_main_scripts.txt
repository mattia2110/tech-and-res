# State Finder Script

ztr_state_finder_script = {
	ordered_scope_state = {
		limit = {
			NOT = { is_homeland_of_country_cultures = root }
			state_region = {
				is_homeland = scope:ztr_minor_primary_culture
			}
		}
	order_by = gdp
	save_scope_as = ztr_state_finder
	}
}

# Religions Script

african_religion_setting = {
	if = {
		limit = {
			religion_percent_country = {
				target = rel:catholic
				value >= 0.5
			}
		}
		set_state_religion = rel:catholic
	}
	else_if = {
		limit = {
			religion_percent_country = {
				target = rel:protestant
				value >= 0.5
			}
		}
		set_state_religion = rel:protestant
	}
	else_if = {
		limit = {
			religion_percent_country = {
				target = rel:sunni
				value >= 0.5
			}
		}
		set_state_religion = rel:sunni
	}
	else = {
		set_state_religion = rel:sunni
	}
}

# There are some laws that will cap the maximum liberty index a country can have
calculate_max_liberty_index = {	

	if = {
		limit = {
			var:liberty_index_var > 35
			has_law_or_variant = law_type:law_single_party_state
		}
		remove_variable = liberty_index_var
		set_variable = { name = liberty_index_var value = 35 }
	}
	else_if = {
		limit = {
			var:liberty_index_var > 45
			has_law_or_variant = law_type:law_autocracy
		}
		remove_variable = liberty_index_var
		set_variable = { name = liberty_index_var value = 45 }		
	}
	else_if = {
		limit = {
			var:liberty_index_var > 49
			has_law_or_variant = law_type:law_oligarchy
		}
		remove_variable = liberty_index_var
		set_variable = { name = liberty_index_var value = 49 }		
	}
	else_if = {
		limit = {
			var:liberty_index_var > 55
			OR = {
				has_law_or_variant = law_type:law_technocracy
				has_law_or_variant = law_type:law_landed_voting
				has_law_or_variant = law_type:law_wealth_voting
			}
		}
		remove_variable = liberty_index_var
		set_variable = { name = liberty_index_var value = 55 }		
	}
	else_if = {
		limit = {
			var:liberty_index_var > 65
			has_law_or_variant = law_type:law_census_voting
		}
		remove_variable = liberty_index_var
		set_variable = { name = liberty_index_var value = 65 }		
	}

	if = {
		limit = {
			var:liberty_index_var > 100
		}
		remove_variable = liberty_index_var
		set_variable = { name = liberty_index_var value = 100 }		
	}
	else_if = {
		limit = {
			var:liberty_index_var < 0
		}
		remove_variable = liberty_index_var
		set_variable = { name = liberty_index_var value = 0 }
	}
}
