on_yearly_pulse = {
	on_actions = {
		ztr_main_script_on_yearly_pulse
		ztr_decolonization_age_on_yearly_pulse

ztr_main_script_on_yearly_pulse = {
	effect = {
		# Special buildings creation on capital states
		every_country = {
			limit = {
				has_technology_researched = modern_urban_planning
				NOT = { has_building = building_modern_state_baseline }
			}
			capital = {
				create_building = {
					building = "building_modern_state_baseline"
					level = 1
				}
			}
		}

		# Special buildings removal from non-capital states
		every_country = {
			limit = {
				any_scope_state = {
					is_capital = no
					OR = {
						has_building = building_research_center
						has_building = building_fusion_power_plant
						has_building = building_modern_state_baseline
					}
					count > 0
				}
			}
			every_scope_state = {
				limit = {
					is_capital = no
					OR = {
						has_building = building_research_center
						has_building = building_fusion_power_plant
						has_building = building_modern_state_baseline
					}
				}
				remove_building = building_research_center
				remove_building = building_fusion_power_plant
				remove_building = building_modern_state_baseline
			}
		}

		# Early progressist ideologies
		every_country = {
			limit = {
				NOT = { has_variable = has_progressist_ideology }
				has_technology_researched = human_rights
			}
			set_variable = {
				name = has_progressist_ideology
				value = 1
			}
			ztr_apply_early_progressist_ideologies = yes
		}

		# Mid progressist ideologies
		every_country = {
			limit = {
				has_variable = has_progressist_ideology
				var:has_progressist_ideology = 1
				has_technology_researched = socialdemocracy
			}
			change_variable = {
				name = has_progressist_ideology
				add = 1
			}
			ztr_apply_mid_progressist_ideologies = yes
		}

		# Late progressist ideologies
		every_country = {
			limit = {
				has_variable = has_progressist_ideology
				var:has_progressist_ideology = 2
				has_technology_researched = neoliberism
			}
			change_variable = {
				name = has_progressist_ideology
				add = 1
			}
			ztr_apply_late_progressist_ideologies = yes
		}
	}
}

ztr_decolonization_age_on_yearly_pulse = {
	effect = {
		if = {
			limit = {
				has_game_rule = allow_ztr_decolonization_content
				any_country = {
					has_technology_researched = decolonization
				}
				NOT = { has_global_variable = ztr_global_decolonization_pressure_var }
			}

			# First time: initialize the global variable and announce the decolonization age
			set_global_variable = {
				name = ztr_global_decolonization_pressure_var
				value = 0
			}
			every_country = {
				limit = {
					country_rank >= rank_value:minor_power
				}
				trigger_event = { id = ztr_decolonization_age.2 }
			}
		}
		else_if = {
			limit = {
				has_game_rule = allow_ztr_decolonization_content
				has_global_variable = ztr_global_decolonization_pressure_var
			}

			# Subsequent years: recalculate global decolonization pressure
			ztr_calculate_global_decolonization_pressure = yes
		}
	}
}

un_institutions_on_yearly_pulse = {
	trigger = {
		has_global_variable = un_founded_var
	}
	effect = {
		# ICC global mechanics
		every_country = {
			limit = {
				has_global_variable = ztr_un_institution_icc
				global_var:ztr_un_institution_icc = 1
				ztr_is_un_member = yes
				infamy >= 25
			}
			trigger_event = { id = ztr_un_event.45 }
		}

		# IMF global mechanics
		# IMF poor countries management
		every_country = {
			limit = {
				has_global_variable = ztr_un_institution_imf
				ztr_is_un_member = yes
				OR = {
					average_sol < 10
					has_imf_modifier = yes
				}
			}
			remove_imf_modifier = yes
			if = {
				limit = {
					average_sol < 6
				}
				add_modifier = extreme_imf_help_modifier
			}
			else_if = {
				limit = {
					average_sol < 7
				}
				add_modifier = big_imf_help_modifier
			}
			else_if = {
				limit = {
					average_sol < 8
				}
				add_modifier = medium_imf_help_modifier
			}
			else_if = {
				limit = {
					average_sol < 9
				}
				add_modifier = small_imf_help_modifier
			}
			else_if = {
				limit = {
					average_sol < 10
				}
				add_modifier = tiny_imf_help_modifier
			}
		}
	}
}

atomic_on_yearly_pulse_country = {
	events = {
		nuclear_warfare_system.2
	}
	effect = {
		# Clear nuclear weapons silo if no longer have nuclear program
		if = {
			limit = {
				NOT = { has_modifier = nuclear_capability }
				NOT = { has_variable = nuclear_program_started }
				any_scope_state = {
					any_scope_building = {
						OR = {
							is_building_type = building_nuclear_weapons_silo
						}
					}
				}
			}

			# Clear nuclear buildings
			every_scope_state = {
				limit = { has_building = building_nuclear_weapons_silo }
				remove_building = building_nuclear_weapons_silo
			}
		}

		# Clean up nuclear weapons program variable
		if = {
			limit = {
				has_variable = nuclear_program_started
				OR = {
					NOT = { has_law = law_type:law_uranium_full_usage }
					has_variable = signed_nuclear_non_proliferation_treaty
					is_subject = yes
				}
			}
			remove_variable = nuclear_program_started
		}
	}
}

				limit = {
					NOT = { has_global_variable = ztr_hidden_traits_var }
