ztr_un_country_can_propose_possible_law = {
	has_modifier = ztr_onu_permanent_member
	has_global_variable = ztr_un_subject_possible_proposal
	has_global_variable = ztr_un_law_possible_proposal
	NOT = {
		has_global_variable = ztr_un_vote_in_progress
	}

	OR = {
		AND = {
			global_var:ztr_un_subject_possible_proposal = 1
			OR = {
				global_var:ztr_un_law_possible_proposal = 1
				global_var:ztr_un_law_possible_proposal = 2
				global_var:ztr_un_law_possible_proposal = 3
				global_var:ztr_un_law_possible_proposal = 4
			}
		}
		AND = {
			global_var:ztr_un_subject_possible_proposal = 2
			OR = {
				global_var:ztr_un_law_possible_proposal = 0
				global_var:ztr_un_law_possible_proposal = 1
			}
		}
		AND = {
			global_var:ztr_un_subject_possible_proposal = 3
			OR = {
				global_var:ztr_un_law_possible_proposal = 1
				global_var:ztr_un_law_possible_proposal = 2
				global_var:ztr_un_law_possible_proposal = 3
