# Trigger is altered by liberty index and not multiculturalism laws since they already modify culture acceptance score values.
#TODO Migliorare questo trigger con condizioni più specifiche che considerano fervor, continente, tolleranza etc...
culture_wants_to_be_decolonized = {
	root = {
		hidden_trigger = { has_variable = liberty_index_var }
		OR = {
			AND = {
				ztr_is_full_democratic = yes
				cultural_acceptance_base = {
					target = prev
					value <= 55 #acceptance_status_3 #40
				}
			}
			AND = {
				ztr_is_flawed_democratic = yes
				cultural_acceptance_base = {
					target = prev
					value <= 60 #acceptance_status_3 #40
				}
			}
			AND = {
				ztr_is_not_democratic = yes
				cultural_acceptance_base = {
					target = prev
					value <= 65 #acceptance_status_5 #80
				}
			}
		}
	}
}

culture_wants_to_be_decolonized_incorporated = {
	root = {
		hidden_trigger = { has_variable = liberty_index_var }
		OR = {
			AND = {
				ztr_is_full_democratic = yes
				cultural_acceptance_base = {
					target = prev
					value <= 45 #acceptance_status_3 #40
				}
			}
			AND = {
				ztr_is_flawed_democratic = yes
				cultural_acceptance_base = {
					target = prev
					value <= 50 #acceptance_status_3 #40
				}
			}
			AND = {
				ztr_is_not_democratic = yes
				cultural_acceptance_base = {
					target = prev
					value <= 55 #acceptance_status_5 #80
				}
			}
		}
	}
}

is_colonial_power = {
	custom_tooltip = {
		text = ztr_has_subject_or_decolonizing_colonial_state_tt
		OR = {
			# Exists a subject
			any_subject_or_below = {
				count > 0
			}
			# Exists a state with homeland cultures that want to be decolonized
			any_scope_state = {
				state_want_to_be_decolonized = yes
			}
		}
	}
}

state_want_to_be_decolonized = {
	NOT = { is_homeland_of_country_cultures = root }
	any_scope_culture = {
		filter = {
			has_homeland = prev
		}
		count > 0
		always = yes
	}
	OR = {
		AND = {
			is_incorporated = yes
			any_scope_culture = {
				filter = {
					has_homeland = prev
				}
				count = all
				culture_wants_to_be_decolonized_incorporated = yes
			}
		}
		AND = {
			is_incorporated = no
			any_scope_culture = {
				filter = {
					has_homeland = prev
				}
				count = all
				culture_wants_to_be_decolonized = yes
			}
		}
	}
}

###########################################################
#----------PHASE-CHECK TRIGGERS FOR DECOLONIZATION--------#
###########################################################

# Phase 1: Quiet Phase (0+)
ztr_decolonization_is_phase_1 = {
	has_variable = ztr_decolonization_pressure_var
	var:ztr_decolonization_pressure_var >= 0
}

# Phase 2: Request Phase (30+)
ztr_decolonization_is_phase_2 = {
	has_variable = ztr_decolonization_pressure_var
	var:ztr_decolonization_pressure_var >= 30
}

# Phase 3-4: Revolt Phase (50+)
ztr_decolonization_is_phase_3 = {
	has_variable = ztr_decolonization_pressure_var
	var:ztr_decolonization_pressure_var >= 50
}

# Phase 4: Full Revolt Phase (75+)
ztr_decolonization_is_phase_4 = {
	has_variable = ztr_decolonization_pressure_var
	var:ztr_decolonization_pressure_var >= 75
}
