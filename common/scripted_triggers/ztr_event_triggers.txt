# Flavor Triggers

# Trigger is altered by liberty index and not multiculturalism laws since they already modify culture acceptance score values.
wants_to_be_decolonized = {
		root = {
			hidden_trigger = { has_variable = liberty_index_var }
			OR = {
				AND = {
					ztr_is_full_democratic = yes
					cultural_acceptance_base = {
						target = prev
						value < acceptance_status_4
					}
				}
				AND = {
					ztr_is_flawed_democratic = yes
					cultural_acceptance_base = {
						target = prev
						value <= acceptance_status_4
					}
				}
				AND = {
					ztr_is_not_democratic = yes
					cultural_acceptance_base = {
						target = prev
						value <= acceptance_status_5
					}
				}
			}
		}
}

hungary_is_subject_or_personal_union_austria = {
	c:AUS ?= this
	exists = c:HUN
	c:HUN = {
		OR = {
			overlord = root
			has_diplomatic_pact = {
				who = root
				type = personal_union
				is_initiator = no
			}
		}
		has_state_in_state_region = STATE_TRANSDANUBIA
	}
}

# Building Triggers

has_research_center = {
	any_scope_state = {
		any_scope_building = {
			is_building_type = building_research_center
			is_under_construction = no
		}
	}
}

has_research_center_target = {
	any_scope_building = {
		is_building_type = building_research_center
		is_under_construction = no
	}
}
