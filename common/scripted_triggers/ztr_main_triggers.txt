# Building Triggers

has_research_center = {
	any_scope_state = {
		any_scope_building = {
			is_building_type = building_research_center
			is_under_construction = no
		}
	}
}

has_research_center_target = {
	any_scope_building = {
		is_building_type = building_research_center
		is_under_construction = no
	}
}

has_fusion_reactor_target = {
	any_scope_building = {
		is_building_type = building_fusion_power_plant
		is_under_construction = no
	}
}

# Similar Index Triggers

ztr_is_very_similar_liberty_index = {
	OR = {
		AND = {
			scope:actor = {
				ztr_is_full_democratic = yes
				ztr_is_communist = no
			}
			scope:target_country = {
				ztr_is_full_democratic = yes
				ztr_is_communist = no
			}
		}
		AND = {
			scope:actor = {
				ztr_is_totalitarian = yes
				ztr_is_communist = no
			}
			scope:target_country = {
				ztr_is_totalitarian = yes
				ztr_is_communist = no
			}
		}
	}
}

ztr_similar_liberty_index = {
	OR = {
		AND = {
			scope:actor = {
				ztr_is_democratic = yes
			}
			scope:target_country = {
				ztr_is_democratic = yes
			}
		}
		AND = {
			scope:actor = {
				ztr_is_not_democratic = yes
			}
			scope:target_country = {
				ztr_is_not_democratic = yes
			}
		}
		AND = {
			scope:actor = {
				ztr_is_communist = yes
			}
			scope:target_country = {
				ztr_is_communist = yes
			}
		}
	}
}

ztr_neutral_liberty_index = {
	OR = {
		AND = {
