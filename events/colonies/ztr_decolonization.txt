# Winds of Decolonization
ztr_decolonization.1 = {
	dlc = dlc_tech_res
	type = country_event
	placement = ROOT

	event_image = {
		trigger = {
			scope:subject_scope = { country_is_in_europe = yes }
		}
		video = "europenorthamerica_springtime_of_nations"

		on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	}
	event_image = {
		trigger = {
			scope:subject_scope = { country_is_in_north_africa = yes }
		}
		video = "africa_leader_arguing"

		on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	}
	event_image = {
		trigger = {
			scope:subject_scope = { country_is_in_middle_east = yes }
		}
		video = "middleeast_courtroom_upheaval"

		on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	}
	event_image = {
		trigger = {
			scope:subject_scope = { country_is_in_subsaharan_africa = yes }
		}
		video = "africa_public_protest"

		on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	}
	event_image = {
		trigger = {
			scope:subject_scope = { country_is_in_india = yes }
		}
		video = "ip2_india_protest"

		on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	}
	event_image = {
		trigger = {
			scope:subject_scope = { country_is_in_east_asia = yes }
		}
		video = "ep1_donghak_rebellion"

		on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	}
	event_image = {
		trigger = {
			scope:subject_scope = { country_is_in_southeast_asia = yes }
		}
		video = "asia_buddhism"

		on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	}
	event_image = {
		trigger = {
			scope:subject_scope = {
				NOR = {
					country_is_in_europe = yes
					country_is_in_north_africa = yes
					country_is_in_middle_east = yes
					country_is_in_subsaharan_africa = yes
					country_is_in_india = yes
					country_is_in_east_asia = yes
					country_is_in_southeast_asia = yes
				}
			}
		}
		video = "southamerica_war_civilians"
		on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	}

	icon = "gfx/interface/icons/event_icons/decolonization_event.dds"

	title = ztr_decolonization.1.t
	desc = ztr_decolonization.1.d


	duration = 3
	
	trigger = {
		has_game_rule = allow_ztr_decolonization_content
		has_global_variable = ztr_decolonization_age_var
		any_subject_or_below = {
			#No decolonization events for british raj vassals, as they are too much and very annoying
			NOT = { is_subject_of = c:BIC}
			NOT = {	c:BIC = root }
			#No decolonization events for west indies vassals, as they are too much and very annoying
			NOT = { is_subject_of = c:DEI}
			NOT = {	is_diplomatic_play_enemy_of = root }
			NOT = {	has_truce_with = root }
			liberty_desire > 90
		}
	}

	immediate = {
		set_variable = {
			name = decolonization_subject_event_var
			value = 0
		}
		set_variable = {
			name = decolonization_subject_event_var_neg
			value = 0
		}
		random_subject_or_below = {
			limit = {
				#No decolonization events for british raj vassals, as they are too much and very annoying
				NOT = { is_subject_of = c:BIC}
				NOT = {	c:BIC = root }
				#No decolonization events for west indies vassals, as they are too much and very annoying
				NOT = { is_subject_of = c:DEI}
				NOT = {	is_diplomatic_play_enemy_of = root }
				NOT = {	has_truce_with = root }
				liberty_desire > 90
			}
			save_scope_as = subject_scope
		}
		calculate_infamy_decolonization_subject_event = yes
		set_variable = {
			name = decolonization_subject_event_var_neg
			value = var:decolonization_subject_event_var
		}
		change_variable = {
			name = decolonization_subject_event_var_neg
			multiply = -1
		}
	}

	option = {
		name = ztr_decolonization.1.a
		ai_chance = {
			base = 10
			# 4:1 for full democracies, 2.5:1 for democracies and 1:1 for others
			modifier = {
				trigger = {
					ztr_is_full_democratic = yes
				}
				add = 15
			}
			modifier = {
				trigger = {
					ztr_is_democratic = yes
				}
				add = 15
			}
		}
		create_bidirectional_truce = { country = scope:subject_scope months = 120 }
		change_infamy = var:decolonization_subject_event_var_neg
		remove_variable = decolonization_subject_event_var
		remove_variable = decolonization_subject_event_var_neg
		scope:subject_scope = {
			make_independent = yes
			change_relations = {
				country = root
				value = 50
			}
		}
	}

	option = {
		name = ztr_decolonization.1.b
		default_option = yes
		ai_chance = {
			base = 10
		}
		create_incident = {
			country = scope:subject_scope
			value = var:decolonization_subject_event_var
		}
		remove_variable = decolonization_subject_event_var
		remove_variable = decolonization_subject_event_var_neg
		scope:subject_scope = {
			change_relations = {
				country = root
				value = -100
			}
			create_diplomatic_play = {
			target_country = root
			type = dp_independence
		}
		}
	}
}

ztr_decolonization.11 = {
	dlc = dlc_tech_res
	type = country_event
	placement = ROOT
	event_image = {
		video = "southamerica_war_civilians"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/decolonization_event.dds"

	title = ztr_decolonization.11.t
	desc = ztr_decolonization.11.d


	duration = 3
	
	trigger = {
		has_game_rule = allow_ztr_decolonization_content
		cu:bedouin = { wants_to_be_decolonized = yes }
		has_global_variable = ztr_decolonization_age_var
		year > 1945
		NOT = { is_subject = yes }
		NOT = {
			country_has_primary_culture = cu:bedouin
		}
		NOT = {
			OR = {
				c:BHN ?= root
				c:ABU ?= root
			}
		}
		OR = {
			s:STATE_ABU_DHABI = {
			any_scope_state = {
					owner = ROOT
				}
			}
		}
	}

	immediate = {
	}

	option = {
		name = ztr_decolonization.11.a
		ai_chance = {
			base = 90
		}
		change_infamy = -5
		if = {
			limit = {
				NOT = {
					exists = c:BHN
				}
			}
			create_country = {
			tag = BHN
			origin = this
			province = p:x505AE0
			on_created = {
				set_country_type = recognized
				activate_law = law_type:law_theocracy
			}
			}
		}
		if = {
			limit = {
				NOT = {
					exists = c:ABU
				}
			}
			create_country = {
			tag = ABU
			origin = this
			province = p:x65a581
			on_created = {
				set_country_type = recognized
				activate_law = law_type:law_theocracy
			}
			}
		}
		if = {
			limit = {
				s:STATE_ABU_DHABI = {
				any_scope_state = {
						owner = ROOT
					}
				}
			}
			s:STATE_ABU_DHABI = {
