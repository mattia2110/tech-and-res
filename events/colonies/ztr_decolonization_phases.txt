# Event 1: The Educated Native (Phase 1, Homeland)
# A Western-educated native intellectual publishes a treatise demanding self-governance
ztr_decolonization_phases.1 = {
	dlc = dlc_tech_res
	type = country_event
	placement = scope:decolonizing_state

	event_image = {
		video = "africa_leader_arguing"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/decolonization_event.dds"

	title = ztr_decolonization_phases.1.t
	desc = ztr_decolonization_phases.1.d

	duration = 3

	trigger = {
		has_game_rule = allow_ztr_decolonization_content
		ztr_decolonization_is_phase_1 = yes
		NOT = { has_variable = ztr_decol_educated_native_cd }
		any_scope_state = {
			state_want_to_be_decolonized = yes
			literacy_rate > 0.15
		}
		ztr_is_not_totalitarian = yes
	}

	immediate = {
		set_variable = {
			name = ztr_decol_educated_native_cd
			years = 10
		}
		random_scope_state = {
			limit = {
				state_want_to_be_decolonized = yes
				literacy_rate > 0.15
			}
			save_scope_as = decolonizing_state
		}
	}

	option = {
		name = ztr_decolonization_phases.1.a
		ai_chance = {
			base = 10
			modifier = {
				trigger = { ztr_is_democratic = yes }
				add = 15
			}
		}
		change_variable = {
			name = ztr_decolonization_pressure_var
			add = 2
		}
	}

	option = {
		name = ztr_decolonization_phases.1.b
		default_option = yes
		ai_chance = {
			base = 10
		}
		change_infamy = 1
		scope:decolonizing_state = {
			add_radicals_in_state = {
				value = small_radicals
			}
		}
	}
}
