# Event 1: The Overlord's Slaves
# The UN slavery ban forces the metropole to confront pre-colonial slavery
# maintained by native overlords in colonial territories as a status quo
ztr_decolonization_slavery.1 = {
	dlc = dlc_tech_res
	type = country_event
	placement = scope:overlord_slavery_state

	event_image = {
		video = "africa_leader_arguing"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/decolonization_event.dds"

	title = ztr_decolonization_slavery.1.t
	desc = ztr_decolonization_slavery.1.d

	duration = 3

	trigger = {
		has_game_rule = allow_ztr_decolonization_content
		has_global_variable = ztr_un_hr_slavery
		global_var:ztr_un_hr_slavery >= 1
		ztr_is_un_member = yes
		NOT = { has_variable = ztr_decol_slavery_overlord_cd }
		has_law = law_type:law_slavery_banned
		any_scope_state = {
			state_want_to_be_decolonized = yes
		}
		is_colonial_power = yes
	}

	immediate = {
		set_variable = {
			name = ztr_decol_slavery_overlord_cd
			years = 15
		}
		random_scope_state = {
			limit = {
				state_want_to_be_decolonized = yes
			}
			save_scope_as = overlord_slavery_state
		}
	}

	option = {
		name = ztr_decolonization_slavery.1.a
		ai_chance = {
			base = 10
			modifier = {
				trigger = { ztr_is_democratic = yes }
				add = 20
			}
		}
		change_variable = {
			name = ztr_decolonization_pressure_var
			add = 3
		}
		change_infamy = -3
		# Native elites resist the universal enforcement
		every_scope_state = {
			limit = {
				state_want_to_be_decolonized = yes
			}
			add_radicals_in_state = {
				value = small_radicals
			}
		}
	}

	option = {
		name = ztr_decolonization_slavery.1.b
		default_option = yes
		ai_chance = {
			base = 10
		}
		change_infamy = 3
		every_scope_state = {
			limit = {
				state_want_to_be_decolonized = yes
			}
			add_loyalists_in_state = {
				value = small_loyalists
			}
		}
	}
}

	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/decolonization_event.dds"

	title = ztr_decolonization_slavery.2.t
	desc = ztr_decolonization_slavery.2.d

	duration = 3

	trigger = {
		has_game_rule = allow_ztr_decolonization_content
		has_global_variable = ztr_un_hr_slavery
		global_var:ztr_un_hr_slavery >= 1
		ztr_is_un_member = yes
		NOT = { has_variable = ztr_decol_slavery_liberation_fired }
		has_law = law_type:law_slavery_banned
		has_variable = ztr_recently_banned_slavery
		any_scope_state = {
			state_want_to_be_decolonized = yes

		}

		# One-time event

			limit = {

			}

		}
	}

	option = {
		name = ztr_decolonization_slavery.2.a
		ai_chance = {
			base = 10
			modifier = {
				trigger = { ztr_is_democratic = yes }
				add = 15
			}
		}
		add_treasury = -50000
		every_scope_state = {
			limit = {
				state_want_to_be_decolonized = yes
			}
			add_loyalists_in_state = {
				value = small_loyalists
			}
		}
	}

	option = {
		name = ztr_decolonization_slavery.2.b
		default_option = yes
		ai_chance = {
			base = 10
		}
		every_scope_state = {
			limit = {
				state_want_to_be_decolonized = yes
			}
			add_radicals_in_state = {
				value = medium_radicals
			}
		}
	}
}

		name = ztr_decolonization_slavery.3.a
		ai_chance = {
			base = 10
			modifier = {
				trigger = { ztr_is_democratic = yes }
				add = 15
			}
		}
		change_variable = {
			name = ztr_decolonization_pressure_var
			add = 2
		}
		add_authority = -100
		scope:defiance_state = {
			add_loyalists_in_state = {
				value = small_loyalists
			}
		}
	}

	option = {
		name = ztr_decolonization_slavery.3.b
		default_option = yes
		ai_chance = {
			base = 10
		}
		change_infamy = 1
		scope:defiance_state = {
			add_radicals_in_state = {
				value = small_radicals
			}
		}
	}
}
