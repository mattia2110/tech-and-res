namespace = ztr_superpower

ztr_superpower.1 = {
	dlc = dlc_tech_res
	type = country_event
	placement = root

	duration = 1

	title = ztr_superpower.1.t
	desc = ztr_superpower.1.d

	event_image = {
		video = "unspecific_military_parade"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/bicorne_hat.dds"

	trigger = {
		global_country_ranking <= 2
		has_technology_researched = neoimperialism
		NOT = { has_modifier = ztr_superpower }
	}

	immediate = {
	}

	#Add modern medicine modifier
	option = {
		name = ztr_superpower.1.a
		default_option = yes
        hidden_effect = {
            if = {
                limit = {
                    has_modifier = ztr_superpower
                }
                remove_modifier = ztr_superpower
            }
        }
		add_modifier = ztr_superpower
	}
}

	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/bicorne_hat.dds"

	trigger = {
		has_modifier = ztr_superpower
		has_law_or_variant = law_type:law_national_militia
	}

	immediate = {
		law_type:law_professional_army = {
			save_scope_as = professional_law_scope
		}
		law_type:law_mass_conscription = {
			save_scope_as = conscription_law_scope
		}
	}

	#Accept Professional Army
	option = {
		name = ztr_superpower.professional.option
		default_option = yes
		activate_law = law_type:law_professional_army
	}

	#Accept Mass Conscription
	option = {
		name = ztr_superpower.mass.option
		default_option = yes
		activate_law = law_type:law_mass_conscription
	}

	#Decline
	option = {
		name = ztr_superpower.decline.option
	}
