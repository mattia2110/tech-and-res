namespace = ztr_decolonization

ztr_decolonization.1 = {
	dlc = dlc_tech_res
	type = country_event
	placement = ROOT
	event_image = {
		video = "southamerica_war_civilians"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	title = ztr_decolonization.1.t
	desc = ztr_decolonization.1.d


	duration = 3
	
	trigger = {
		has_game_rule = allow_ztr_decolonization_content
		has_global_variable = ztr_decolonization_age_var
		any_country = {
			is_subject_of = ROOT
			NOT = { is_subject_of = c:BIC} #No decolonization events for british raj vassals, as they are too much and very annoying
			NOT = { is_subject_of = c:DEI} #No decolonization events for west indies vassals, as they are too much and very annoying
			NOT = {	is_diplomatic_play_enemy_of = ROOT }
			NOT = {	has_truce_with = ROOT }
			NOT = {	c:BIC = THIS }
			# NOT = { 
			# 	OR = {
			# 		c:BIC = THIS
			# 		c:DEI = THIS
			# 	}
			# }
			liberty_desire > 90
		}
	}

	immediate = {
		random_country = {
			limit = {
				is_subject_of = ROOT
				NOT = { is_subject_of = c:BIC} #No decolonization events for british raj vassals, as they are too much and very annoying
				NOT = { is_subject_of = c:DEI} #No decolonization events for west indies vassals, as they are too much and very annoying
				NOT = { is_diplomatic_play_enemy_of = ROOT }
				NOT = { has_truce_with = ROOT }
				NOT = {	c:BIC = THIS }
				# NOT = {
				# 	OR = {
				# 		c:BIC = THIS
				# 		c:DEI = THIS
				# 	}
				# }
				liberty_desire > 90
			}
			save_scope_as = subject_scope
		}
	}

	option = {
		name = ztr_decolonization.1.a
		ai_chance = {
			base = 90
		}
		create_bidirectional_truce = { country = scope:subject_scope months = 120 }
		change_infamy = -2
		scope:subject_scope = {
			make_independent = yes
			change_relations = {
				country = root
				value = 50
			}
		}
	}

	option = {
		name = ztr_decolonization.1.b
		default_option = yes
		ai_chance = {
			base = 10
		}
		change_infamy = 5
		scope:subject_scope = {
			change_relations = {
				country = root
				value = -100
			}
			create_diplomatic_play = {
			target_country = root
			type = dp_independence
		}
		}
	}
}

	title = ztr_decolonization.2.t
	desc = ztr_decolonization.2.d


	duration = 3
	
	trigger = {
		has_game_rule = allow_ztr_decolonization_content
		has_global_variable = ztr_decolonization_age_var
		year > 1945
		NOT = { is_subject = yes }
		NOT = {
			OR = {
				c:TUN ?= root
			}
		}
		NOT = {
			country_has_primary_culture = cu:maghrebi
		}
		OR = {
			country_is_in_europe = yes
			country_is_in_north_america = yes
			country_is_in_middle_east = yes
		}
		s:STATE_TUNISIA = {
			any_scope_state = {
				owner = ROOT
			}
		}
		NOT = {
			c:ITA ?= root
		}
	}

	immediate = {
	}

	option = {
		name = ztr_decolonization.2.a
		ai_chance = {
			base = 90
		}
		change_infamy = -5
		if = {
			limit = {
				NOT = {
				exists = c:TUN
			}
			}
			create_country = {
			tag = TUN
			origin = this
			province = p:x80D0A0
			on_created = {
				set_country_type = recognized
				activate_law = law_type:law_parliamentary_republic
				activate_law = law_type:law_autocracy
			}
			}
		}
		s:STATE_TUNISIA = {
			random_scope_state = {
				limit = {
					owner = root
				}
			set_state_owner = c:TUN
			}
		}
		create_bidirectional_truce = { country = c:TUN months = 120 }
		c:TUN = {
			change_relations = {
				country = root
				value = 50
			}
		}
	}

	option = {
		name = ztr_decolonization.2.b
		default_option = yes
		ai_chance = {
			base = 10
		}
		change_infamy = 5

		}

		}

		}
		create_bidirectional_truce = { country = c:CHI months = 120 }
		c:CHI = {
			change_relations = {
				country = root
				value = 50
			}
		}

	}

	option = {
		name = ztr_decolonization.21.b
		default_option = yes
		ai_chance = {
			base = 10
		}
		change_infamy = 10
	}
