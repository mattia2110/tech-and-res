namespace = ztr_culture_obsession

# Cosmetic Obsession (French)

ztr_culture_obsession.1 = {
	dlc = dlc_tech_res
	type = country_event
	placement = ROOT
	event_image = {
		video = "unspecific_world_fair"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	title = ztr_culture_obsession.1.t
	desc = ztr_culture_obsession.1.d


	duration = 1
	
	trigger = {
		OR = {
			AND = {
				exists = c:FRA
				c:FRA = this
				NOT = { has_global_variable = ztr_cosmetic_french_obsession }
			}
			OR = {
				exists = c:WAL
				c:WAL = this
				NOT = { has_global_variable = ztr_cosmetic_wallonian_obsession }
			}
		}
		
	}

	immediate = {
		
	}
	
	option = {
		name = ztr_culture_obsession.1.a
		default_option = yes
		trigger = {
			c:FRA = this
		}
		set_global_variable = {
			name = ztr_cosmetic_french_obsession
			value = 1
		}
		cu:french = { add_cultural_obsession = cosmetics }
		ai_chance = {
			base = 2										
		}		
	}

	option = {
		name = ztr_culture_obsession.1.b
		trigger = {
			c:BEL = this
		}
		set_global_variable = {
			name = ztr_cosmetic_wallonian_obsession
			value = 1
		}
		cu:wallonian = { add_cultural_obsession = cosmetics }
		ai_chance = {
			base = 2										
		}		
	}
}
