# Egypt Sinai-Suez Crisis (Egypt/Arabian Power wants European powers to leave Suez Canal)

ztr_middle_east.1 = {
	dlc = dlc_tech_res
	type = country_event
	placement = root

	duration = 1

	title = ztr_middle_east.1.t
	desc = ztr_middle_east.1.d

	event_image = {
		video = "africa_construction_colony"
	}

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	trigger = {
		year > 1950
		has_global_variable = ztr_decolonization_age_var
		has_game_rule = allow_ztr_decolonization_content
		NOT = { is_subject = yes }
		OR = {
			country_is_in_africa = yes
			country_is_in_middle_east = yes
		}
		OR = {
			has_state_in_state_region = STATE_LOWER_EGYPT
			has_state_in_state_region = STATE_UPPER_EGYPT
			has_state_in_state_region = STATE_MIDDLE_EGYPT
		}
		s:STATE_SINAI ?= {
			has_building = building_suez_canal
		}
		NOT = { has_global_variable = ztr_suez_crisis_var }
		scope:ztr_european_colonial_power_scope = {
			NOT = { has_war_with = ROOT }
		}
	}

	immediate = {
		every_country = {
			limit = {
				has_state_in_state_region = STATE_SINAI
				country_is_in_europe = yes
			}
			save_scope_as = ztr_european_colonial_power_scope
		}
		root = {
			save_scope_as = ztr_arabian_country_scope
		}
		set_global_variable = ztr_suez_crisis_var
	}

	# Arabian Country can ask them to leave peacefully with a financial purchase of 9 millions
	option = {
		name = ztr_middle_east.1.a
		add_loyalists = {
			value   = 0.5
			pop_type = officers
		}
		add_loyalists = {
			value   = 0.25
			pop_type = soldiers
		}
		add_loyalists = {
			value   = 0.1
			culture = cu:misri
		}
		scope:ztr_european_colonial_power_scope = { trigger_event = ztr_middle_east.2 }
		ai_chance = {
			base = 7
			modifier = {
				trigger = {
					scaled_debt > 0 # Is in debt
				}
				subtract = 7
			}
		}
	}

	# Arabian Country can avoid any problem
	option = {
		name = ztr_middle_east.1.b
		default_option = yes
		add_radicals = {
			value   = 0.75
			pop_type = officers
		}
		add_radicals = {
			value   = 0.5
			pop_type = soldiers
		}
		add_radicals = {
			value   = 0.15
			culture = cu:misri
		}
		ai_chance = {
			base = 3
		}
	}
}

ztr_middle_east.2 = {

	type = country_event
	placement = root

	duration = 1

	title = ztr_middle_east.2.t
	desc = ztr_middle_east.2.d

	event_image = {
		video = "africa_construction_colony"
	}

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	trigger = {
		# Middle_east.1
	}

	immediate = {
		every_scope_state = {
			limit = {
				state_region = s:STATE_SINAI
			}
			save_scope_as = ztr_sinai_state_european
		}
	}

	# Accept diplomatic solution with money concession of 90 millions
	# Obviously, the Arabian Country will pay less for this.

	option = {
		name = ztr_middle_east.2.a
		scope:ztr_arabian_country_scope = {
			hidden_effect= {
				add_modifier = {
					name = suez_canal_big_money_exit
					months = 36
					is_decaying = no
				}
			}
		}
		add_modifier = {
			name = suez_canal_big_money
			months = 36
			is_decaying = no
		}
		custom_tooltip = {
			text = middle_east_suez_canal_tt
			every_scope_state = {
				limit = {
					state_region = s:STATE_SINAI
				}
				set_state_owner = scope:ztr_arabian_country_scope
			}
		}
		ai_chance = {
			base = 3
			modifier = {
				trigger = {
					scope:ztr_arabian_country_scope = {
						relations:root >= relations_threshold:cordial
					}
				}
				add = 7
			}
		}
	}

	# Neutral Option, giving some of the Suez Canal earning

	option = {
		name = ztr_middle_east.2.b
		# This modifiers are permanent
		add_modifier = {
			name = suez_canal_small_money_exit
			is_decaying = no
		}
		scope:ztr_arabian_country_scope = {
			add_modifier = {
				name = suez_canal_small_money
				is_decaying = no
			}
		}
		ai_chance = {
			base = 3
			modifier = {
				trigger = {
					scope:ztr_arabian_country_scope = {
						relations:root >= relations_threshold:neutral
					}
				}
				add = 7
			}
		}
	}

	# Refuse diplomatic

	option = {
		name = ztr_middle_east.2.c
		scope:ztr_arabian_country_scope = {
			change_relations = {
				country = root
				value = -100
			}
		}
		scope:ztr_arabian_country_scope = {
			create_diplomatic_play = {
				name = annex_war
				target_country = root
				type = dp_return_state
				target_state = scope:ztr_sinai_state_european
				add_war_goal = { holder = scope:ztr_arabian_country_scope target_country = root 
				type = enforce_treaty_article
				article = {
						article = money_transfer
						source_country = root
						target_country = scope:ztr_arabian_country_scope
						inputs = {
							{ quantity = 50000 }
						}
					}
				}
			}
		}
		ai_chance = {
			base = 10
		}
	}

}
