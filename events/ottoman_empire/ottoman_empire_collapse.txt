            limit = {
                any_scope_state = {
                    region = sr:region_arabic

                limit = {
                    region = sr:region_arabic
                }
                save_scope_as = arab_congress_state
            }

        capital = { save_scope_as = ztr_ottoman_capital_scope }
    }

    option = {
        name = ztr_ottoman_collapse.2.a

        change_variable = {
            name = ottoman_empire_collapse_var
            subtract = 2

        if = {
            limit = { var:ottoman_empire_collapse_arab_var > 0 }
            change_variable = {
                name = ottoman_empire_collapse_arab_var
                subtract = 1
            }
        }

            value = 0.05
            pop_type = bureaucrats
        }

        ai_chance = {
            base = 3
            modifier = {
                trigger = {
                    OR = {
                        has_law = law_type:law_cultural_exclusion
                        has_law = law_type:law_multicultural

                }
                add = 4
            }
        }
    }

    option = {
        name = ztr_ottoman_collapse.2.b
        default_option = yes

        change_variable = {
            name = ottoman_empire_collapse_var
            add = 3
        }
        change_variable = {
            name = ottoman_empire_collapse_arab_var
            add = 2
        }
        change_variable = {
            name = ottoman_empire_collapse_army_var
            add = 1
        }
        add_radicals = {
            value = 0.05
            pop_type = farmers
        }
        add_radicals = {
            value = 0.05
            pop_type = academics
        }
        add_radicals = {
            value = 0.05
            pop_type = bureaucrats
