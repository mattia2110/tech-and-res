namespace = france_pop

france_pop.1 = {
	dlc = dlc_tech_res
	type = country_event
	placement = root

	duration = 1

	title = france_pop.1.t
	desc = france_pop.1.d

	event_image = {
		video = "unspecific_world_fair"
	}

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	trigger = {
		c:FRA ?= root
		year < 1900
		NOT = { has_variable = historical_population_of_france_var }
		NOT = { has_modifier = historical_population_of_france_1}
	}

	immediate = {
		set_variable = {
			name = historical_population_of_france_var
			value = 0
		}
	}

	option = {
		name = france_pop.1.a
		default_option = yes
		custom_tooltip = france_pop_tt
		hidden_effect = {
			remove_modifier = historical_population_of_france_1
		}
		add_modifier = historical_population_of_france_1
	}
}
