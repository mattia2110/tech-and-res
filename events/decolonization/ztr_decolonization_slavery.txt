# Event 1: The Overlord's Slaves
# The UN slavery ban forces the metropole to confront pre-colonial slavery
# maintained by native overlords in colonial territories as a status quo
ztr_decolonization_slavery.1 = {
	dlc = dlc_tech_res
	type = country_event
	placement = scope:overlord_slavery_state

	event_image = {
		video = "southamerica_slave_chains"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	title = ztr_decolonization_slavery.1.t
	desc = ztr_decolonization_slavery.1.d

	duration = 3

	trigger = {
		has_game_rule = allow_ztr_decolonization_content
		has_global_variable = ztr_un_hr_slavery
		global_var:ztr_un_hr_slavery >= 1
		ztr_is_un_member = yes
		NOT = { has_variable = ztr_decol_slavery_overlord_cd }
		has_law = law_type:law_slavery_banned
		any_scope_state = {
			ztr_state_is_colonial = yes
			NOT = { is_incorporated = yes }
			OR = {
				is_in_geographic_region = geographic_region_africa
				is_in_geographic_region = geographic_region_middle_east
				is_in_geographic_region = geographic_region_india
				is_in_geographic_region = geographic_region_east_asia
				is_in_geographic_region = geographic_region_south_east_asia
				is_in_geographic_region = geographic_region_central_asia
			}
		}
		is_colonial_power = yes
	}

	immediate = {
		random_scope_state = {
			limit = {
				ztr_state_is_colonial = yes
				NOT = { is_incorporated = yes }
				OR = {
					is_in_geographic_region = geographic_region_africa
					is_in_geographic_region = geographic_region_middle_east
					is_in_geographic_region = geographic_region_india
					is_in_geographic_region = geographic_region_east_asia
					is_in_geographic_region = geographic_region_south_east_asia
					is_in_geographic_region = geographic_region_central_asia
				}
			}
			save_scope_as = overlord_slavery_state
		}
	}

	option = {
		name = ztr_decolonization_slavery.1.a
		ai_chance = {
			base = 10
			modifier = {
				trigger = { ztr_is_democratic = yes }
				add = 20
			}
		}
		set_variable = {
			name = ztr_decol_slavery_overlord_cd
			years = 15
		}
		custom_tooltip = ztr_decolonization_pressure_increase_5_tt
		change_variable = {
			name = ztr_decolonization_pressure_var
			add = 5
		}
		change_infamy = -5
		# Native elites resist the universal enforcement
		custom_tooltip = {
			text = ztr_decolonization_slavery_1_a_effect_tt
			every_scope_state = {
				limit = {
					ztr_state_is_colonial = yes
				}
				add_radicals_in_state = {
					value = small_radicals
				}
			}
			every_scope_culture = {
				limit = {
					NOT = { is_primary_culture_of = root }
					any_scope_state = {
						is_homeland = prev
						ztr_state_is_colonial = yes
						count > 0
					}
				}
				add_fervor = 5
			}
		}
	}

	option = {
		name = ztr_decolonization_slavery.1.b
		default_option = yes
		ai_chance = {
			base = 10
		}
		change_infamy = 5
		custom_tooltip = {
			text = ztr_decolonization_slavery_1_b_effect_tt
			every_scope_state = {
				limit = {
					ztr_state_is_colonial = yes
				}
				add_loyalists_in_state = {
					value = small_radicals
				}
			}
		}
		set_variable = {
			name = ztr_decol_slavery_overlord_cd
			years = 15
		}
		if = {
			limit = {
				has_global_variable = ztr_un_institution_icc
				global_var:ztr_un_institution_icc = 1
			}
			add_modifier = {
				name = icc_investigation_modifier
				years = 5
			}
		}
		else = {
			add_modifier = {
				name = ztr_un_condemned_modifier
				years = 3
			}
		}
			
		#TODO Aggiungere modifier che migliora approval landowners di 5 per 120 mesi
	}
}

# Event 2: The Mass Liberation
# Thousands of newly freed slaves flood colonial cities seeking work and shelter
ztr_decolonization_slavery.2 = {
	dlc = dlc_tech_res
	type = country_event
	placement = scope:liberation_state

	event_image = {
		video = "africa_public_protest"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	title = ztr_decolonization_slavery.2.t
	desc = ztr_decolonization_slavery.2.d

	duration = 3

	trigger = {
		has_game_rule = allow_ztr_decolonization_content
		has_global_variable = ztr_un_hr_slavery
		global_var:ztr_un_hr_slavery >= 1
		ztr_is_un_member = yes
		NOT = { has_variable = ztr_decol_slavery_liberation_fired }
		has_law = law_type:law_slavery_banned
		has_variable = ztr_recently_banned_slavery
		any_scope_state = {
			ztr_state_is_colonial = yes
			NOT = { is_incorporated = yes }
			OR = {
				is_in_geographic_region = geographic_region_africa
				is_in_geographic_region = geographic_region_middle_east
				is_in_geographic_region = geographic_region_india
				is_in_geographic_region = geographic_region_east_asia
				is_in_geographic_region = geographic_region_south_east_asia
				is_in_geographic_region = geographic_region_central_asia
			}
			state_population > 50000
		}
	}

	immediate = {
		# One-time event
		set_variable = ztr_decol_slavery_liberation_fired
		random_scope_state = {
			limit = {
				ztr_state_is_colonial = yes
				NOT = { is_incorporated = yes }
				OR = {
					is_in_geographic_region = geographic_region_africa
					is_in_geographic_region = geographic_region_middle_east
					is_in_geographic_region = geographic_region_india
					is_in_geographic_region = geographic_region_east_asia
					is_in_geographic_region = geographic_region_south_east_asia
					is_in_geographic_region = geographic_region_central_asia
				}
				state_population > 50000
			}
			save_scope_as = liberation_state
		}
		define_ztr_gdp_dependent_cost = yes
	}

	option = {
		name = ztr_decolonization_slavery.2.a
		ai_chance = {
			base = 10
			modifier = {
				trigger = { ztr_is_democratic = yes }
				add = 15
			}
			modifier = {
				trigger = { scaled_debt > 0.5 }
				multiply = 0.5
			}
		}
		ztr_gdp_dependent_cost = yes
		custom_tooltip = {
			text = ztr_decolonization_slavery_2_a_effect_tt
			every_scope_state = {
				limit = {
					ztr_state_is_colonial = yes
				}
				add_loyalists_in_state = {
					value = small_radicals
				}
			}
		}
		custom_tooltip = ztr_decolonization_pressure_increase_3_tt
		change_variable = {
			name = ztr_decolonization_pressure_var
			add = 3
		}
	}

	option = {
		name = ztr_decolonization_slavery.2.b
		default_option = yes
		ai_chance = {
			base = 10
		}
		custom_tooltip = {
			text = ztr_decolonization_slavery_2_b_effect_tt
			every_scope_state = {
				limit = {
					ztr_state_is_colonial = yes
				}
				add_radicals_in_state = {
					value = medium_radicals
				}
			}
		}
		custom_tooltip = ztr_decolonization_pressure_increase_8_tt
		change_variable = {
			name = ztr_decolonization_pressure_var
			add = 8
		}
	}
}
