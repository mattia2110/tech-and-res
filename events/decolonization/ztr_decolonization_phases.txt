# Event 1: The Educated Native (Phase 1, Homeland)
# A Western-educated native intellectual publishes a treatise demanding self-governance
ztr_decolonization_phases.1 = {
	dlc = dlc_tech_res
	type = country_event
	placement = scope:decolonizing_state

	event_image = {
		video = "africa_leader_arguing"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	title = ztr_decolonization_phases.1.t
	desc = ztr_decolonization_phases.1.d

	duration = 3

	trigger = {
		has_game_rule = allow_ztr_decolonization_content
		ztr_decolonization_is_phase_1 = yes
		NOT = { has_variable = ztr_decol_educated_native_cd }
		any_scope_state = {
			ztr_state_is_colonial = yes
			NOT = { is_incorporated = yes }
			literacy_rate > 0.15
		}
		ztr_is_not_totalitarian = yes
	}

	immediate = {
		set_variable = {
			name = ztr_decol_educated_native_cd
			years = 10
		}
		random_scope_state = {
			limit = {
				ztr_state_is_colonial = yes
				NOT = { is_incorporated = yes }
				literacy_rate > 0.15
			}
			save_scope_as = decolonizing_state
		}
		scope:decolonizing_state = {
			random_scope_culture = {
				limit = {
					has_homeland = prev
				}
				save_scope_as = native_culture_scope
			}
		}
	}

	option = {
		name = ztr_decolonization_phases.1.a
		ai_chance = {
			base = 10
			modifier = {
				trigger = { ztr_is_democratic = yes }
				add = 15
			}
		}
		custom_tooltip = ztr_decolonization_pressure_increase_8_tt
		change_variable = {
			name = ztr_decolonization_pressure_var
			add = 8
		}
		scope:decolonizing_state = {
			every_scope_culture = {
				limit = {
					has_homeland = prev
				}
				add_fervor = 10
				save_scope_as = native_culture_scope
			}
		}
	}

	option = {
		name = ztr_decolonization_phases.1.b
		default_option = yes
		trigger = {
			OR = {
				has_law_or_variant = law_type:law_outlawed_dissent
				has_law_or_variant = law_type:law_censorship
			}
		}
		ai_chance = {
			base = 10
		}
		change_infamy = 5
		custom_tooltip = ztr_decolonization_pressure_decrease_5_tt
		change_variable = {
			name = ztr_decolonization_pressure_var
			add = -5
		}
		scope:decolonizing_state = {
			add_radicals_in_state = {
				value = small_radicals
			}
		}
	}
}

# Event 2: Missionaries and Conscience (Phase 1, Subject)
# Religious missionaries returning from colonies publish reports exposing poor conditions

	dlc = dlc_tech_res
	type = country_event
	placement = scope:missionary_subject

	event_image = {
		video = "africa_animism"
	}

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	title = ztr_decolonization_phases.2.t
	desc = ztr_decolonization_phases.2.d

	duration = 3

	trigger = {
		has_game_rule = allow_ztr_decolonization_content
		ztr_decolonization_is_phase_1 = yes
		NOT = { has_variable = ztr_decol_missionaries_cd }
		any_subject_or_below = {
			any_scope_state = {
				average_sol < 12
			}
			NOT = { religion = root.religion }
			NOT = { has_variable = ztr_decolonization_subject_cd }
		}
		OR = {
			ig:ig_devout ?= { is_marginal = no }
			ig:ig_rural_folk ?= { is_marginal = no }
		}
	}

	immediate = {
		set_variable = {
			name = ztr_decol_missionaries_cd
			years = 10
		}
		random_subject_or_below = {
			limit = {
				any_scope_state = {
					average_sol < 12
				}
				NOT = { religion = root.religion }
				NOT = { has_variable = ztr_decolonization_subject_cd }
			}
			save_scope_as = missionary_subject
		}
	}

	option = {
		name = ztr_decolonization_phases.2.a
		ai_chance = {
			base = 10
			modifier = {
				trigger = { ztr_is_democratic = yes }
				add = 15
			}
		}
		change_infamy = 2
		if = {
			limit = {
				OR = {
					ig:ig_devout ?= { is_marginal = no }
					ig:ig_rural_folk ?= { is_marginal = no }
				}
			}
			add_loyalists = {
				value = 0.03
				interest_group = ig:ig_devout
			}
			add_loyalists = {
				value = 0.01
				interest_group = ig:ig_rural_folk
			}
		}
		custom_tooltip = ztr_decolonization_pressure_increase_5_tt
		change_variable = {
			name = ztr_decolonization_pressure_var
			add = 5
		}
	}

	option = {
		name = ztr_decolonization_phases.2.b
		default_option = yes
		ai_chance = {
			base = 10
		}
		if = {
			limit = {
				OR = {
					ig:ig_devout ?= { is_marginal = no }
					ig:ig_rural_folk ?= { is_marginal = no }
				}
			}
			add_radicals = {
				value = 0.1
				interest_group = ig:ig_devout
			}
			add_radicals = {
				value = 0.05
				interest_group = ig:ig_rural_folk
			}
		}

# Event 4: Whispers in the Bazaar (Phase 1, Subject)
# Intelligence reports reveal underground nationalist networks forming among colonial subjects
ztr_decolonization_phases.4 = {
	dlc = dlc_tech_res
	type = country_event
	placement = scope:whispers_subject

	event_image = {
		video = "middleeast_courtroom_upheaval"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	title = ztr_decolonization_phases.4.t
	desc = ztr_decolonization_phases.4.d

	duration = 3

	trigger = {
		has_game_rule = allow_ztr_decolonization_content
		ztr_decolonization_is_phase_1 = yes
		NOT = { has_variable = ztr_decol_whispers_cd }
		any_subject_or_below = {
			liberty_desire > 50
			NOT = { has_variable = ztr_decolonization_subject_cd }
			NOT = { is_subject_of = c:BIC }
			can_increase_autonomy = yes
			OR = {
				is_in_geographic_region = geographic_region_africa
				is_in_geographic_region = geographic_region_middle_east
				is_in_geographic_region = geographic_region_india
				is_in_geographic_region = geographic_region_south_east_asia
				is_in_geographic_region = geographic_region_central_asia
			}
		}
		is_at_war = no
	}

	immediate = {
		set_variable = {
			name = ztr_decol_whispers_cd
			years = 10
		}
		random_subject_or_below = {
			limit = {
				liberty_desire > 50
				NOT = { has_variable = ztr_decolonization_subject_cd }
				NOT = { is_subject_of = c:BIC }
				can_increase_autonomy = yes
				OR = {
					is_in_geographic_region = geographic_region_africa
					is_in_geographic_region = geographic_region_middle_east
					is_in_geographic_region = geographic_region_india
					is_in_geographic_region = geographic_region_south_east_asia
					is_in_geographic_region = geographic_region_central_asia
				}
			}
			save_scope_as = whispers_subject
		}
	}

	option = {
		name = ztr_decolonization_phases.4.a
		ai_chance = {
			base = 10
			modifier = {
				trigger = { ztr_is_democratic = yes }
				add = 15
			}
		}
		custom_tooltip = ztr_decolonization_pressure_increase_2_tt
		change_variable = {
			name = ztr_decolonization_pressure_var
			add = 2
		}
		#TODO Aggiungi una perdita di denaro per overlord e modificatore per più bureaucracy e authority per subject
		scope:whispers_subject = {
			add_liberty_desire = {
				value = -50
			}
			increase_autonomy = yes
		}
		change_relations = {
			country = scope:whispers_subject
			value = 10
		}
	}

	option = {
		name = ztr_decolonization_phases.4.b
		default_option = yes
		ai_chance = {
			base = 10
		}
		scope:whispers_subject = {
			add_liberty_desire = {
				value = 15
			}
			if = {
				limit = {
					NOT = { has_law_or_variant = law_type:law_censorship }
				}
				activate_law = law_type:law_censorship
			}
			else_if = {
				limit = {
					NOT = { has_law_or_variant = law_type:law_outlawed_dissent }
				}
				activate_law = law_type:law_outlawed_dissent
			}
		}
		change_relations = {
			country = scope:whispers_subject
			value = -50
		}
	}
}
