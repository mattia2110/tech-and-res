namespace = ztr_imf_event

#IMF foundation
ztr_imf_event.1 = {
	dlc = dlc_tech_res
	type = country_event
	placement = root

	duration = 1

	title = {
		first_valid = {
			triggered_desc = {
				desc = ztr_imf_conference.t1
				trigger = {
					has_state_in_state_region = STATE_NEW_HAMPSHIRE
				}
			}
			triggered_desc = {
				desc = ztr_imf_conference.t2
				trigger = {
					NOT = {
						has_state_in_state_region = STATE_NEW_HAMPSHIRE
					}
				}
			}
		}
	}
	desc = ztr_imf_event.1.d

	event_image = { 
		#texture = "gfx/event_pictures/un_foundation.dds"
		video = "un_foundation"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/ztr_onu_permanent_member.dds"

	trigger = {
		#Decision
	}

	immediate = {
		if = {
			limit = {
				has_state_in_state_region = STATE_NEW_HAMPSHIRE
			}
			random_state = {
				limit = {
					state_region = s:STATE_NEW_HAMPSHIRE
				}
				save_scope_as = imf_conference_state
			}
		}
		else = {
			random_scope_state = {
				limit = {
					is_incorporated = yes
				}
				save_scope_as = imf_conference_state
			}
		}
		root = {
			save_scope_as = imf_proposer
		}
	}

	option = {
		name = ztr_imf_event.1.a
		default_option = yes
		change_infamy = -5
		every_country = {
			limit = {
				NOT = { this = root }
				OR = {
					has_modifier = ztr_onu_member
					has_modifier = ztr_onu_permanent_member
				}
			}
			trigger_event = ztr_imf_event.2
		}
		ai_chance = {
			base = 100
		}
	}
}

#IMF poor countries management
ztr_imf_event.3 = {
	dlc = dlc_tech_res
	type = country_event
	placement = root
	hidden = yes

	trigger = {
		has_global_variable = imf_founded_var
		OR = {
			has_modifier = ztr_onu_member
			has_modifier = ztr_onu_permanent_member
		}
		OR = {
			average_sol < 10
			has_imf_modifier = yes
		}
	}

	immediate = {
		#Depending on SoL, the country receives help from the IMF.
		remove_imf_modifier = yes
		if = {
			limit = {
				average_sol < 6
			}
			add_modifier = extreme_imf_help_modifier
		}
		else_if = {
			limit = {
				average_sol < 7
			}
			add_modifier = big_imf_help_modifier
		}
		else_if = {
			limit = {
				average_sol < 8
			}
			add_modifier = medium_imf_help_modifier
		}
		else_if = {
			limit = {
				average_sol < 9
			}
			add_modifier = small_imf_help_modifier
		}
		else_if = {
			limit = {
				average_sol < 10
			}
			add_modifier = tiny_imf_help_modifier
		}
	}
}
