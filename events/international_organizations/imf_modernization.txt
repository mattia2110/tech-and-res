# Brain Drain Crisis (Domestic)
ztr_imf_modernization.10 = {
	dlc = dlc_tech_res
	type = country_event
	placement = root
	duration = 1
	title = ztr_imf_modernization.10.t
	desc = ztr_imf_modernization.10.d
	flavor = ztr_imf_modernization.10.f
	event_image = {
		video = "asia_poor_people_moving"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/generic_icons/international_organizations.dds"
	trigger = {
		literacy_rate < 0.5
		any_country = {
			is_country_type = recognized
			country_rank >= rank_value:great_power
			NOT = {
				this = root
			}
			average_sol > root.average_sol
		}
		NOT = {
			has_variable = imf_brain_drain_cooldown
		}
	}
	immediate = {
		set_variable = {
			name = imf_brain_drain_cooldown
			years = 8
		}
		random_country = {
			limit = {
				is_country_type = recognized
				country_rank >= rank_value:great_power
				NOT = {
					this = root
				}
				average_sol > root.average_sol
			}
			save_scope_as = brain_drain_destination
		}
	}
	# Fund university scholarships
	option = {
		name = ztr_imf_modernization.10.a
		default_option = yes
		add_modifier = {
			name = imf_scholarship_fund_modifier
			years = 10
		}
		ordered_scope_state = {
			order_by = gdp
			max = 1
			create_building = {
				building = building_university
				level = 1
			}
			if = {
				limit = {
					prev = {
						has_technology_researched = telephone
					}
				}
				create_building = {
					building = building_electrics_industry
					level = 1
				}
			}
			else = {
				create_building = {
					building = building_electronics_industry
					level = 1
				}
			}
		}
		ai_chance = {
			base = 25
			modifier = {
				trigger = {
					literacy_rate < 0.35
				}
				add = 20
			}
			modifier = {
				trigger = {
					average_sol >= 12
				}
				add = 10
			}
			modifier = {
				trigger = {
					scaled_debt < 0.25
				}
				add = 10
			}
		}
	}
	# Restrict emigration
	option = {
		name = ztr_imf_modernization.10.b
		add_modifier = {
			name = imf_emigration_restriction_modifier
			years = 10
		}
		ai_chance = {
			base = 15
			modifier = {
				trigger = {
					ztr_is_not_democratic = yes
				}
				add = 20
			}
			modifier = {
				trigger = {
					scaled_debt > 0.25
				}
				add = 10
			}
			modifier = {
				trigger = {
					ig:ig_intelligentsia = {
						is_powerful = no
					}
				}
				add = 10
			}
		}
	}
	# Ignore the problem
	option = {
		name = ztr_imf_modernization.10.c
		add_modifier = {
			name = imf_brain_drain_modifier
			years = 10
		}
		ai_chance = {
			base = 5
			modifier = {
				trigger = {
					get_country_gdp_per < 2.5
				}
				add = 10
			}
			modifier = {
				trigger = {
					scaled_debt > 0.5
				}
				add = 10
			}
			modifier = {
				trigger = {
					average_sol < 8
				}
				add = 5
			}
		}
	}
