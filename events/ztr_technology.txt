# Fusion Reactor - Purchase request from another country
ztr_technology.14 = {
	type = country_event
	title = fusion_reactor_purchase.1.t
	desc = fusion_reactor_purchase.1.d
	event_image = {
		video = "quantum_physics"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/nuclear_fusion_reactor.dds"
	
	option = {
		name = fusion_reactor_purchase.1.a
		default_option = yes
		# Accept request
		scope:fusion_technology_requester = {
			trigger_event = {
				id = ztr_technology.15
				popup = yes
			}
		}
		ai_chance = {
			base = 100
			modifier = {
				trigger = {
					relations:scope:fusion_technology_requester >= relations_threshold:amicable
				}
				add = 100
			}
			# Command Hegemony see this as a threat
			modifier = {
				trigger = {
					has_strategy = ai_strategy_superpower_command_hegemony
				}
				add = -100
			}
			# Requester Great Powers are less likely to receive accepted requests if they are not friendly
			modifier = {
				trigger = {
					scope:fusion_technology_requester = {
						country_rank = rank_value:great_power
					}
					relations:scope:fusion_technology_requester < relations_threshold:friendly
				}
				add = -50
			}
		}
	}

	option = {
		name = fusion_reactor_purchase.1.b
		# Reject request
		scope:fusion_technology_requester = {
			trigger_event = {
				id = ztr_technology.16
				popup = yes
			}
		}
		ai_chance = {
			base = 100
		}
	}
}

# Fusion Reactor - Purchase request accepted
ztr_technology.15 = {
	type = country_event
	title = fusion_reactor_purchase.2.t
	desc = fusion_reactor_purchase.2.d
	event_image = {
		video = "quantum_physics"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	icon = "gfx/interface/icons/event_icons/nuclear_fusion_reactor.dds"
	duration = 3
	immediate = {
		# Massive cost representing technology licensing and infrastructure import
		add_modifier = {
			name = fusion_reactor_purchase_cost
			months = 60
		}
		# We add 30 million pounds to the seller
		scope:fusion_technology_seller = {
			add_treasury = 30000000
		}
		# Apply site modifier directly to capital (proven technology, no survey needed)
		capital = {
			add_modifier = {
				name = fusion_reactor_site
			}
			save_scope_as = fusion_purchase_site
		}
		add_technology_researched = nuclear_fusion_reactor
		# Start construction JE directly
		add_journal_entry = {
			type = je_fusion_reactor_construction
			target = scope:fusion_purchase_site
		}
	}
	option = {
		name = fusion_reactor_purchase.2.a
		default_option = yes
	}
}
