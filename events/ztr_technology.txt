# Fusion Reactor - Construction complete
ztr_technology.11 = {
	type = country_event

	title = fusion_reactor.2.t

	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					NOT = { has_global_variable = ztr_fusion_reactor_first_var }
				}
				desc = fusion_reactor.2.d_first
			}
			triggered_desc = {
				trigger = {
					has_global_variable = ztr_fusion_reactor_first_var
				}
				desc = fusion_reactor.2.d
			}
		}
	}

	placement = scope:fusion_reactor_state

	event_image = {
		video = "quantum_physics"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/innovation.dds"

	duration = 3

	trigger = {
		# triggered by je_fusion_reactor_construction
	}

	immediate = {
		save_scope_as = fusion_builder_country
		random_scope_state = {
			limit = {
				has_building = building_fusion_power_plant
			}
			save_scope_as = fusion_reactor_state
			random_scope_building = {
				limit = {
					is_building_type = building_fusion_power_plant
				}
				save_scope_as = fusion_reactor_building_scope
			}
		}
	}

	option = {
		name = fusion_reactor.2.a_first
		trigger = {
			NOT = { has_global_variable = ztr_fusion_reactor_first_var }
		}
		default_option = yes
		# Mark this as the first fusion reactor ever built
		set_global_variable = ztr_fusion_reactor_first_var
		# Prestige bonus for being the pioneer
		add_modifier = fusion_energy_pioneer
		# Notify all other countries
		every_country = {
			limit = { NOT = { this = root } }
			trigger_event = { id = ztr_technology.13 }
		}
	}

	option = {
		name = fusion_reactor.2.a
		trigger = {
			has_global_variable = ztr_fusion_reactor_first_var
		}
		default_option = yes
	}
}
