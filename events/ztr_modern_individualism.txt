namespace = ztr_modern_individualism

# Paradigm shift: the Industrial ideology moves from classical Individualism
# to Modern Individualism (post-Keynesian / social-liberal / early neoliberal).
ztr_modern_individualism.1 = {
	dlc = dlc_tech_res
	type = country_event
	placement = ROOT
	event_image = {
		video = "ztr_developing_country"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/economy_update.dds"

	title = ztr_modern_individualism.1.t
	desc = ztr_modern_individualism.1.d
	flavor = ztr_modern_individualism.1.f

	duration = 3

	trigger = {
		get_country_gdp_per > 5
		literacy_rate >= 0.5
		has_technology_researched = keynesian_economics
		exists = ig:ig_industrialists
		NOT = { ztr_is_communist = yes }
		NOT = { enacting_any_law = yes }
		ig:ig_industrialists ?= {
			has_ideology = ideology:ideology_individualist
		}
	}

	immediate = {
		ig:ig_industrialists ?= {
			save_scope_as = industrialist_ig_scope
		}
	}

	option = {
		name = ztr_modern_individualism.1.a
		default_option = yes
		ig:ig_industrialists ?= {
			remove_ideology = ideology_individualist
			add_ideology = ideology_modern_individualism
		}
		if = {
			limit = {
				enacting_any_law = no
				NOT = { has_law_or_variant = law_type:law_old_age_pension }
			}
			start_enactment = law_type:law_old_age_pension
		}
		ai_chance = {
			base = 100
		}
	}

	option = {
		name = ztr_modern_individualism.1.b
		ig:ig_industrialists ?= {
			remove_ideology = ideology_individualist
			add_ideology = ideology_modern_individualism
		}
		ai_chance = {
			base = 10
		}
	}
}
