Button: ztr_un_open_human_rights_menu
menu_event = ztr_un_event.30
subject_event = ztr_un_event.31 | subject_id = 4 | subject = Decolonization | variable = ztr_un_hr_decolonization
  0 = Neutral -> no forced law change
  1 = Conservative Decolonization -> activate law_no_colonial_affairs from law_colonial_resettlement or law_colonial_exploitation
  2 = Encouraged Decolonization -> activate law_no_colonial_affairs from law_frontier_colonization, law_colonial_resettlement, or law_colonial_exploitation
subject_event = ztr_un_event.32 | subject_id = 5 | subject = Slavery | variable = ztr_un_hr_slavery
  0 = Negligible / Neutral -> no forced law change
  1 = Slavery Illegal -> activate law_slavery_banned from law_colonial_slavery, law_debt_slavery, law_slave_trade, or law_legacy_slavery
subject_event = ztr_un_event.33 | subject_id = 6 | subject = Children's Rights | variable = ztr_un_hr_children_rights
  0 = Negligible -> no forced law change
  1 = Restricted Child Labor -> activate law_restricted_child_labor from law_child_labor_allowed
  2 = Compulsory Primary School -> activate law_compulsory_primary_school from law_child_labor_allowed or law_restricted_child_labor
  current_init_default = 1 (the nearby comment still says 0 = Negligible)
subject_event = ztr_un_event.34 | subject_id = 7 | subject = Women's Rights | variable = ztr_un_hr_women_rights
  0 = Negligible -> no forced law change
  1 = Gender Equality -> activate law_women_in_the_workplace from law_no_womens_rights or law_women_own_property
